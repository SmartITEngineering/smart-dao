050    /**
051     * This registrar is responsible for scanning and containing configuration of
052     * all resources. Whenever a class is looked up whose package has not been
053     * scanned yet, registrar will scanClassForConfig its package to generate cofiguration and
054     * while generating it will keep scanning until it reaches the leaves, i.e. in
055     * this case association to an object that isn't {@link ResourceDomain}
056     * @author imyousuf
057     * @since 0.4
058     */
059    public class DefaultAnnotationConfigScanner
060        implements ClassConfigScanner,
061                   PackageConfigScanner {
062    
063        private static final String GETTER_PREFIX = "get";
064        private static DefaultAnnotationConfigScanner registrar;
065    
066        static {
067            ConfigRegistrar.registerClassScanner(
068                DefaultAnnotationConfigScanner.class, 25);
069            ConfigRegistrar.registerPackageScanner(
070                DefaultAnnotationConfigScanner.class, 25);
071        }
072    
073        public static DefaultAnnotationConfigScanner getInstance() {
074            if (registrar == null) {
075                registrar = new DefaultAnnotationConfigScanner();
076            }
077            return registrar;
078        }
079        protected final ClassScanner classScanner;
080        protected final Map<Class, EximResourceConfig> configuraitons;
081        protected final Collection<Class> scannedClasses;
082        protected final Collection<String> scannedPackages;
083        protected final ResourceVisitCallback resourceVisitCallback;
084        protected final VisitCallback<AnnotationConfig> callbackHandler;
085    
086        protected DefaultAnnotationConfigScanner() {
087            classScanner = IOFactory.getDefaultClassScanner();
088            configuraitons = new HashMap<Class, EximResourceConfig>();
089            scannedClasses = new HashSet<Class>();
090            scannedPackages = new HashSet<String>();
091            resourceVisitCallback = new ResourceVisitCallback();
092            callbackHandler = resourceVisitCallback;
093        }
094    
095        public synchronized Map<Class, EximResourceConfig> getConfigurations() {
096            return configuraitons;
097        }
098    
099        public synchronized Collection<Class> getConfiguredResourceClasses() {
100            return configuraitons.keySet();
101        }
102    
103        /**
104         * Retrieves the configuration for the resource class, if its not already
105         * generated then it will search the package of the class and generate its
106         * configuraiton.
107         * @param resourceClass The class to scanClassForConfig retrieve configuration for
108         * @return Configuration for the resource; NULL if no resource is available
109         *          or generateable
110         * @throws IllegalArgumentException If resource class is null!
111         */
112        public synchronized EximResourceConfig getResourceConfigForClass(
113            final Class resourceClass) {
114            if (resourceClass == null) {
115                throw new IllegalArgumentException("Resource class can't be null!");
116            }
117            if (configuraitons.containsKey(resourceClass)) {
118                return configuraitons.get(resourceClass);
119            }
120            else {
121                String packageName = resourceClass.getPackage().getName();
122                EximResourceConfig resourceConfig = null;
123                if (!scannedPackages.contains(packageName)) {
124                    resourceConfig = scanPackage(Package.getPackage(packageName),
125                        resourceClass);
126                }
127                return resourceConfig;
128            }
129        }
130    
131        /**
132         * Scans and prepares all configurations in the package and makes it
133         * available for future use.
134         * @param resourcePackage Package to scan and gather configuration
135         * @throws IllegalArgumentException If package is null
136         */
137        public synchronized void scanPackageForResourceConfigs(
138            final Package resourcePackage) {
139            if (resourcePackage == null) {
140                throw new IllegalArgumentException("Resource class can't be null!");
141            }
142            scanPackage(resourcePackage, null);
143        }
144    
145        protected String getPropertyNameFromMethodName(final String methodName) {
146            StringBuilder propertyNameBuilder =
147                new StringBuilder(methodName);
148            propertyNameBuilder.delete(0, GETTER_PREFIX.length());
149            char firstChar = propertyNameBuilder.charAt(0);
150            propertyNameBuilder.delete(0, 1);
151            propertyNameBuilder.insert(0, Character.toUpperCase(firstChar));
152            return propertyNameBuilder.toString();
153        }
154    
155        /**
156         * Scan among the class's annotations to find the required configuration for
157         * exporting and importing resources.
158         * @param probableResourceClass The probable resource domain class.
159         * @return The configuration of the class, null if not annotated with 
160         *          {@link ResourceDomain}
161         */
162        protected EximResourceConfig scanClassForConfig(
163            final Class probableResourceClass) {
164            if (scannedClasses.contains(probableResourceClass)) {
165                return null;
166            }
167            Annotation annotation = probableResourceClass.getAnnotation(
168                ResourceDomain.class);
169            if (annotation == null) {
170                return null;
171            }
172            EximResourceConfigImpl resourceConfig = new EximResourceConfigImpl();
173            resourceConfig.setDomainClass(probableResourceClass);
174            Name nameAnnotation = (Name) probableResourceClass.getAnnotation(
175                Name.class);
176            if (nameAnnotation != null) {
177                resourceConfig.setName(nameAnnotation.value());
178            }
179            else {
180                resourceConfig.setName(probableResourceClass.getName());
181            }
182            ResourceDomain domainAnnotation = (ResourceDomain) annotation;
183            resourceConfig.setAccessByPropertyEnabled(domainAnnotation.
184                accessByProperty());
185            resourceConfig.setAssociateExportPolicyAsUri(domainAnnotation.
186                exportAsURIByDefault());
187            resourceConfig.setPathToResource(domainAnnotation.path());
188            resourceConfig.setExporterImplemented(DomainSelfExporter.class.
189                isAssignableFrom(probableResourceClass));
190            resourceConfig.setImporterImplemented(DomainSelfImporter.class.
191                isAssignableFrom(probableResourceClass));
192            resourceConfig.setIdentityCustomizerImplemented(
193                IdentityCustomizer.class.isAssignableFrom(probableResourceClass));
194            scanMembers(resourceConfig, probableResourceClass);
195            scannedClasses.add(probableResourceClass);
196            configuraitons.put(probableResourceClass, resourceConfig);
197            return resourceConfig;
198        }
199    
200        /**
201         * Scan a package for extracting configurations of resource domains.
202         * @param packageName Package to scanClassForConfig.
203         * @param resourceClass Main class scanClassForConfig requested for
204         * @return The configuration of the resource class, Null if the class is not
205         *          a domain class or resourceClass is null
206         * @throws java.lang.IllegalArgumentException If package is null
207         */
208        protected EximResourceConfig scanPackage(final Package packageToScan,
209                                                 final Class resourceClass)
210            throws IllegalArgumentException {
211            if (packageToScan == null) {
212                throw new IllegalArgumentException();
213            }
214            EximResourceConfig resourceConfig = null;
215            classScanner.scan(new String[]{packageToScan.getName()},
216                new ClassAnnotationVisitorImpl(callbackHandler,
217                ResourceDomain.class.getSimpleName()));
218            Set<String> classPaths = resourceVisitCallback.getProbableResources();
219            if (!classPaths.isEmpty()) {
220                for (String classPath : classPaths) {
221                    try {
222                        Class probableResourceClass =
223                            Class.forName(classPath.replaceAll("/", "\\."));
224                        EximResourceConfig config = scanClassForConfig(
225                            probableResourceClass);
226                        if (config != null && resourceClass != null &&
227                            probableResourceClass.equals(resourceClass)) {
228                            resourceConfig = config;
229                        }
230                    }
231                    catch (Exception ex) {
232                    }
233                }
234            }
235            scannedPackages.add(packageToScan.getName());
236            return resourceConfig;
237        }
238    
239        /**
240         * It will scan all member attributes and behavior based on configuration on
241         * the class. It will also scan all inherited attributes and behavior.
242         * @param resourceConfig The config representing the domain class
243         * @param resourceClass The domain class
244         */
245        protected void scanMembers(final EximResourceConfigImpl resourceConfig,
246                                   final Class resourceClass) {
247            if (resourceConfig.isAccessByPropertyEnabled()) {
248                scanMethods(resourceConfig, resourceClass);
249            }
250            else {
251                scanFields(resourceConfig, resourceClass);
252            }
253        }
254    
255        /**
256         * Scans getter methods for discovering associations of the domain and thier
257         * respective configurations. It will only scan public getter methods.
258         * @param resourceConfig The config of the domain resource
259         * @param resourceClass The domain class
260         */
261        protected void scanMethods(final EximResourceConfigImpl resourceConfig,
262                                   final Class resourceClass) {
263            Method[] methods = resourceClass.getMethods();
264            if (methods == null || methods.length <= 0) {
265                return;
266            }
267            for (Method method : methods) {
268                String methodName = method.getName();
269                //Only scan getter methods as of bean spec, that getter methods with
270                //no paratmeters and non-void return types
271                if (methodName.startsWith(GETTER_PREFIX) && methodName.length() >
272                    GETTER_PREFIX.length() && method.getReturnType() != null &&
273                    !method.getReturnType().equals(Void.class) &&
274                    (method.getParameterTypes() == null ||
275                    method.getParameterTypes().length <= 0)) {
276                    scanGetterMethod(resourceConfig, method);
277                }
278            }
279        }
280    
281        /**
282         * Scans a getter method for annotations which is used to cofigure the
283         * nature of the export
284         * @param resourceConfig The config to populate with configurations
285         * @param method The getter method to scan
286         * @throws java.lang.IllegalArgumentException If its not a getter method
287         *                                            with non-void return type and
288         *                                            with a non-zero length bean
289         *                                            name length and with no
290         *                                            parameter
291         */
292        protected void scanGetterMethod(final EximResourceConfigImpl resourceConfig,
293                                        final Method method)
294            throws IllegalArgumentException {
295            String methodName = method.getName();
296            if (!(methodName.startsWith(GETTER_PREFIX) && methodName.length() >
297                GETTER_PREFIX.length() && method.getReturnType() != null &&
298                !method.getReturnType().equals(Void.class) &&
299                (method.getParameterTypes() == null ||
300                method.getParameterTypes().length <= 0))) {
301                throw new IllegalArgumentException();
302            }
303            String propertyName = getPropertyNameFromMethodName(methodName);
304            Class returnType = method.getReturnType();
305            scanAnnotatedElement(method, propertyName, returnType, resourceConfig);
306        }
307    
308        /**
309         * Scans fields for discovering associations of the domain and thier
310         * respective configurations
311         * @param resourceConfig The config of the domain resource
312         * @param resourceClass The domain class
313         */
314        protected void scanFields(EximResourceConfigImpl resourceConfig,
315                                  Class resourceClass) {
316            Field[] fields = resourceClass.getDeclaredFields();
317            for (Field field : fields) {
318                scanField(resourceConfig, field);
319            }
320            Class parentClass = resourceClass.getSuperclass();
321            if (!parentClass.equals(Object.class)) {
322                scanFields(resourceConfig, parentClass);
323            }
324        }
325    
326        /**
327         * A field is scanned for gathering configurations for export-import.
328         * @param resourceConfig Configuraton for the field association
329         * @param field The field to scan
330         */
331        protected void scanField(final EximResourceConfigImpl resourceConfig,
332                                 final Field field) {
333            String propertyName = field.getName();
334            Class propertyType = field.getType();
335            scanAnnotatedElement(field, propertyName, propertyType, resourceConfig);
336        }
337    
338        /**
339         * Scan an ennotated element to extract configuration information
340         * @param element Element to scan
341         * @param propertyName The name of the property scanning
342         * @param propertyType The type of the property
343         * @param resourceConfig The configuration for the domain class
344         * @throws java.lang.IllegalArgumentException If any argument is null
345         */
346        protected void scanAnnotatedElement(final AnnotatedElement element,
347                                            final String propertyName,
348                                            final Class propertyType,
349                                            final EximResourceConfigImpl resourceConfig)
350            throws IllegalArgumentException {
351            if (element == null || propertyName == null || propertyType == null ||
352                resourceConfig == null) {
353                throw new IllegalArgumentException();
354            }
355            Export annotation =
356                element.getAnnotation(Export.class);
357            AssociationConfigImpl configImpl =
358                new AssociationConfigImpl();
359            configImpl.setName(propertyName);
360            configImpl.setAssociationType(AssociationConfig.AssociationType.
361                getAssociationType(propertyType));
362            Eager eager =
363                element.getAnnotation(Eager.class);
364            configImpl.setStringProviderImplemented(StringValueProvider.class.
365                isAssignableFrom(propertyType));
366            configImpl.setEagerSet(eager != null);
367            if (annotation != null) {
368                configImpl.setItToBeExportedAsUri(!annotation.asObject());
369                configImpl.setTransient(annotation.isTransient());
370            }
371            else {
372                configImpl.setItToBeExportedAsUri(false);
373                configImpl.setTransient(false);
374            }
375            resourceConfig.getAssociationConfigs().put(propertyName, configImpl);
376            Id id =
377                element.getAnnotation(Id.class);
378            if (id != null) {
379                resourceConfig.setIdPropertyName(propertyName);
380            }
381        }
382    
383        /**
384         * The visitor callback to be notified for dmain classes.
385         */
386        protected static class ResourceVisitCallback
387            implements VisitCallback<AnnotationConfig> {
388    
389            private Set<String> probableResources = new HashSet<String>();
390    
391            public void handle(AnnotationConfig config) {
392                probableResources.add(config.getClassName());
393            }
394    
395            public Set<String> getProbableResources() {
396                return probableResources;
397            }
398        }
