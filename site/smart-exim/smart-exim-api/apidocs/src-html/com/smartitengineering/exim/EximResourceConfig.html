019    package com.smartitengineering.exim;
020    
021    import com.smartitengineering.domain.exim.DomainSelfExporter;
022    import com.smartitengineering.domain.exim.DomainSelfImporter;
023    import com.smartitengineering.domain.exim.IdentityCustomizer;
024    import java.util.Map;
025    
026    /**
027     * This API represents the configuration of a class. Two configuration objects
028     * are considered to be equal if they represent the same class.
029     * @author imyousuf
030     * @since 0.4
031     */
032    public interface EximResourceConfig {
033    
034        /**
035         * Returns the configuration map for the associations of this resource 
036         * object class. Since every type in Java extends {@link Object} we consider
037         * every member in a resource class to be an association and thus store its
038         * configuration in thus manner. Name of the property is the key to the map.
039         * @return A property name keyed configuration map of all the associations
040         */
041        Map<String, AssociationConfig> getAssociationConfigs();
042    
043        /**
044         * The resource class it generated this configuration from
045         * @return Class this configuration represents
046         */
047        Class getDomainClass();
048    
049        /**
050         * Returns the property name of the ID property of the domain. Resource
051         * objects must either provide a ID property or implement 
052         * {@link IdentityCustomizer}. Otherwise the config generator might choose
053         * to throw an {@link IllegalArgumentException}.
054         * @return Name of the ID property or NULL if it implements {@link IdentityCustomizer}
055         */
056        String getIdPropertyName();
057    
058        /**
059         * Returns the prefix to the ID and suffix to the path to resource. It will
060         * add '/' in between path and prefix if required.
061         * @return Prefix to the ID.
062         */
063        String getIdPrefix();
064        
065        /**
066         * Return name to be used for exporting, e.g., root node element. It can be
067         * overriden in an association by {@link AssociationConfig}'s name.
068         * @return Name to be used for export; not null.
069         */
070        String getName();
071    
072        /**
073         * Return the path to the resource.
074         * @return Path to the resource
075         */
076        String getPathToResource();
077    
078        /**
079         * Setting for testing whether to access members by property accessors or
080         * field. If access is set by property and a setter is not available that it
081         * will be ignored.
082         * @return True if access is to done by getter/setter and false if
083         *          reflection is to be used for setting property value.
084         */
085        boolean isAccessByPropertyEnabled();
086    
087        /**
088         * Configuration for specifying export policy of this object when is in
089         * association. 
090         * @return True if the resource domain to be exported as URI else inline.
091         */
092        boolean isAssociateExportPolicyAsUri();
093    
094        /**
095         * Configuration whether the object is capable to exporting itself or not.
096         * @return True if it implements {@link DomainSelfExporter}
097         */
098        boolean isExporterImplemented();
099    
100        /**
101         * Defines whether the resource class implements its own ID customizer. It
102         * also signifies that it implements its own URI provider.
103         * @return True if it implements {@link IdentityCustomizer}
104         * @see EximResourceConfig#getIdPropertyName() 
105         */
106        boolean isIdentityCustomizerImplemented();
107    
108        /**
109         * Defines whether the resource has its own importer defined for some
110         * media types.
111         * @return True if it implements {@link DomainSelfImporter} else false
112         */
113        boolean isImporterImplemented();
114    }
