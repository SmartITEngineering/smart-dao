035    /**
036     * A central registrar for configuration scanners. It will register scanners
037     * according their priority first and then in lexical order by class name. And
038     * it will also iterate in the same order.
039     * @author imyousuf
040     * @since 0.4
041     */
042    public final class ConfigRegistrar {
043    
044                    private static final Set<ConfigScannerRegistryItem<ClassConfigScanner>> CLASS_CONFIG_SCANNER =
045                                                                                                                                                                                                                                                                                                                    Collections.synchronizedSortedSet(
046                                                    new TreeSet<ConfigScannerRegistryItem<ClassConfigScanner>>());
047                    private static final Set<ConfigScannerRegistryItem<PackageConfigScanner>> PACKAGE_CONFIG_SCANNER =
048                                                                                                                                                                                                                                                                                                                            Collections.synchronizedSortedSet(
049                                                    new TreeSet<ConfigScannerRegistryItem<PackageConfigScanner>>());
050    
051                    static Set<ConfigScannerRegistryItem<ClassConfigScanner>> getClassConfigScannerRegistryItems() {
052                                    return CLASS_CONFIG_SCANNER;
053                    }
054    
055                    static Set<ConfigScannerRegistryItem<PackageConfigScanner>> getPackageConfigScannerRegistryItems() {
056                                    return PACKAGE_CONFIG_SCANNER;
057                    }
058    
059                    /**
060                     * The following static block loads the default config scanner impl classes,
061                     * so that if they have any static initializers then they can invoked
062                     */
063                    static {
064                                    String packageName =
065                                                             ConfigRegistrar.class.getPackage().getName().concat(".impl");
066                                    ClassScanner scanner = IOFactory.getDefaultClassScanner();
067                                    scanner.scan(new String[] {packageName},
068                                                                    new ClassInstanceVisitorImpl(IOFactory.getClassNameForVisitor(
069                                                                    ClassConfigScanner.class), new VisitCallback<Config>() {
070    
071                                                    public void handle(Config config) {
072                                                                    try {
073                                                                                    IOFactory.getClassFromVisitorName(config.getClassName());
074                                                                    }
075                                                                    catch (Exception ex) {
076                                                                                    ex.printStackTrace();
077                                                                    }
078                                                    }
079                                    }));
080                                    scanner.scan(new String[] {packageName},
081                                                                    new ClassInstanceVisitorImpl(IOFactory.getClassNameForVisitor(
082                                                                    PackageConfigScanner.class), new VisitCallback<Config>() {
083    
084                                                    public void handle(Config config) {
085                                                                    try {
086                                                                                    IOFactory.getClassFromVisitorName(config.getClassName());
087                                                                    }
088                                                                    catch (Exception ex) {
089                                                                                    ex.printStackTrace();
090                                                                    }
091                                                    }
092                                    }));
093                    }
094    
095                    private ConfigRegistrar() {
096                                    throw new AssertionError();
097                    }
098    
099                    /**
100                     * Register the class scanner with its priority
101                     * @param classScannerClass Class scanner
102                     * @param priority Priority of the scanner
103                     * @throws IllegalArgumentException If class scanner is null
104                     */
105                    public static void registerClassScanner(
106                                                    final Class<? extends ClassConfigScanner> classScannerClass,
107                                                    int priority) {
108                                    CLASS_CONFIG_SCANNER.add(new ConfigScannerRegistryItem<ClassConfigScanner>(
109                                                                    classScannerClass, priority));
110                    }
111    
112                    /**
113                     * Register the package scanner with its priority
114                     * @param packageScannerClass Package scanner
115                     * @param priority Priority of the scanner
116                     * @throws IllegalArgumentException If class scanner is null
117                     */
118                    public static void registerPackageScanner(
119                                                    final Class<? extends PackageConfigScanner> packageScannerClass,
120                                                    int priority) {
121                                    PACKAGE_CONFIG_SCANNER.add(new ConfigScannerRegistryItem<PackageConfigScanner>(
122                                                                    packageScannerClass, priority));
123                    }
124    
125                    /**
126                     * Scans (if required) and return the configuration for the resource class
127                     * @param resourceClass Resource class
128                     * @return config for the resource class, null if not config found for the
129                     *          class.
130                     */
131                    public static EximResourceConfig getConfigForClass(final Class resourceClass) {
132                                    for (ConfigScannerRegistryItem<ClassConfigScanner> scanner :
133                                                     CLASS_CONFIG_SCANNER) {
134                                                    EximResourceConfig config = scanner.getInstance().
135                                                                                    getResourceConfigForClass(resourceClass);
136                                                    if (config != null) {
137                                                                    return config;
138                                                    }
139                                    }
140                                    return null;
141                    }
142    
143                    /**
144                     * Simply scan a package to gather all its configurations. This would be
145                     * helpful if an only if the scanner caches the config result.
146                     * @param resourcePackage Package to scan for resources
147                     */
148                    public static void scanPackage(final Package resourcePackage) {
149                                    for (ConfigScannerRegistryItem<PackageConfigScanner> scanner :
150                                                     PACKAGE_CONFIG_SCANNER) {
151                                                    scanner.getInstance().scanPackageForResourceConfigs(resourcePackage);
152                                    }
153                    }
154    
155                    /**
156                     * Return all configurations from a package and in process if requried also
157                     * initiate a scan process
158                     * @param resourcePackage Package to scan
159                     * @return All configurations from the resource package
160                     */
161                    public static Collection<EximResourceConfig> getConfigForPackage(
162                                                    final Package resourcePackage) {
163                                    Set<EximResourceConfig> configs = new HashSet<EximResourceConfig>();
164                                    for (ConfigScannerRegistryItem<PackageConfigScanner> scanner :
165                                                     PACKAGE_CONFIG_SCANNER) {
166                                                    PackageConfigScanner configScanner = scanner.getInstance();
167                                                    configScanner.scanPackageForResourceConfigs(resourcePackage);
168                                                    Collection<Class> configClasses = configScanner.
169                                                                                    getConfiguredResourceClasses();
170                                                    for (Class configClass : configClasses) {
171                                                                    if (resourcePackage.equals(configClass.getPackage())) {
172                                                                                    configs.add(configScanner.getConfigurations().get(
173                                                                                                                    configClass));
174                                                                    }
175                                                    }
176                                    }
177                                    return configs;
178                    }
179    
180                    public static class ConfigScannerRegistryItem<T extends ConfigScanner>
181                                                    implements Comparable<ConfigScannerRegistryItem<T>> {
182    
183                                    private Class<? extends T> configClass;
184                                    private int priority;
185    
186                                    public ConfigScannerRegistryItem(Class<? extends T> configClass,
187                                                                                                                                                                     int priority) {
188                                                    if (configClass == null) {
189                                                                    throw new IllegalArgumentException();
190                                                    }
191                                                    this.configClass = configClass;
192                                                    this.priority = priority;
193                                    }
194    
195                                    @Override
196                                    public boolean equals(Object obj) {
197                                                    if (obj == null) {
198                                                                    return false;
199                                                    }
200                                                    if (getClass() != obj.getClass()) {
201                                                                    return false;
202                                                    }
203                                                    final ConfigScannerRegistryItem<T> other =
204                                                                                                                                                                                             (ConfigScannerRegistryItem<T>) obj;
205                                                    if (this.configClass != other.configClass &&
206                                                                    (this.configClass == null ||
207                                                                     !this.configClass.equals(other.configClass))) {
208                                                                    return false;
209                                                    }
210                                                    if (this.priority != other.priority) {
211                                                                    return false;
212                                                    }
213                                                    return true;
214                                    }
215    
216                                    @Override
217                                    public int hashCode() {
218                                                    int hash = 7;
219                                                    hash =
220                                                    71 * hash +
221                                                    (this.configClass != null ? this.configClass.hashCode() : 0);
222                                                    hash = 71 * hash + this.priority;
223                                                    return hash;
224                                    }
225    
226                                    public int compareTo(ConfigScannerRegistryItem<T> obj) {
227                                                    if (obj == null) {
228                                                                    return 1;
229                                                    }
230                                                    Integer myPriority = Integer.valueOf(priority);
231                                                    Integer otherPriority = Integer.valueOf(obj.priority);
232                                                    int myComp = myPriority.compareTo(otherPriority);
233    
234                                                    if (myComp == 0) {
235                                                                    int classComp = configClass.getName().compareTo(obj.configClass.
236                                                                                                    getName());
237                                                                    return classComp;
238                                                    }
239                                                    else {
240                                                                    return myComp;
241                                                    }
242                                    }
243    
244                                    /**
245                                     * Returns the instance of the scanner.
246                                     * @return If static getInstance() method is implemented then it will
247                                     *          invoke that first else will try to use no-args constructor.
248                                     *          Null if both fails.
249                                     */
250                                    public T getInstance() {
251                                                    T instance = null;
252                                                    try {
253                                                                    Method getInstanceMethod = configClass.getMethod("getInstance",
254                                                                                                    new Class[0]);
255                                                                    if (getInstanceMethod != null &&
256                                                                                    Modifier.isStatic(getInstanceMethod.getModifiers())) {
257                                                                                    instance = (T) getInstanceMethod.invoke(null, new Object[0]);
258                                                                    }
259                                                    }
260                                                    catch (NoSuchMethodException exception) {
261                                                    }
262                                                    catch (InvocationTargetException exception) {
263                                                    }
264                                                    catch (IllegalAccessException exception) {
265                                                    }
266                                                    catch (IllegalArgumentException exception) {
267                                                    }
268                                                    if (instance == null) {
269                                                                    try {
270                                                                                    instance = configClass.newInstance();
271                                                                    }
272                                                                    catch (InstantiationException ex) {
273                                                                    }
274                                                                    catch (IllegalAccessException ex) {
275                                                                    }
276                                                                    catch (RuntimeException ex) {
277                                                                    }
278                                                    }
279                                                    return instance;
280                                    }
281                    }
