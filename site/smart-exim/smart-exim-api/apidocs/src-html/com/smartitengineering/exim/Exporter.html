019    package com.smartitengineering.exim;
020    
021    import java.io.IOException;
022    import java.io.OutputStream;
023    import java.lang.annotation.Annotation;
024    import java.lang.reflect.Type;
025    import java.util.List;
026    import java.util.Map;
027    
028    /**
029     * This API provides SPIs to use it to export objects to a stream of media type
030     * it specifies. It is the responsibility of the API client to invoke it for
031     * appropriate {@link MediaType}. API implementors should explicitly mention the
032     * media type it supports. API implementors should also ensure thread safety.
033     * @author imyousuf
034     * @since 0.4
035     */
036    public interface Exporter {
037    
038        /**
039         * An operation that checks whether the the object type qualified by its
040         * class is supported by this exporter. This operation should be consulted
041         * before invoking/using exportObject.
042         * @param clazz The class quantifying the object type
043         * @param genericType The generic applied to the object, if any
044         * @param annotations Annotations used on the class
045         * @return True if and only if export is supported else false.
046         */
047        boolean isExportSupported(Class clazz,
048                                  Type genericType,
049                                  Annotation[] annotations);
050    
051        /**
052         * It provides the operation to predict the length of the exported content.
053         * @param object The object to export
054         * @param clazz The class of the object
055         * @param genericType The generic used with the object.
056         * @param annotations Annotations qualifying the object's class.
057         * @return -1 if length can not be predicted or else the actual length after
058         *          export
059         */
060        long getAfterExportLength(Object object,
061                                  Class clazz,
062                                  Type genericType,
063                                  Annotation[] annotations);
064    
065        /**
066         * Export the object quantified by clazz and genericType and qualified by
067         * the annotaitons to the outpuStream and use the headers to set meta
068         * informations regarding the object.
069         * @param object The object to export
070         * @param clazz The class of the object
071         * @param genericType The generic used with the object.
072         * @param annotations Annotations qualifying the object's class.
073         * @param outputStream The output stream to write and format the object to.
074         * @param headers The headers that holds the meta information
075         * @throws java.io.IOException If any error in writing to the stream, i.e.
076         *                             either format error or any other error.
077         */
078        void exportObject(Object object,
079                          Class clazz,
080                          Type genericType,
081                          Annotation[] annotations,
082                          OutputStream outputStream,
083                          Map<Object, List<Object>> headers)
084            throws IOException;
085    
086        /**
087         * Provides the {@link MediaType} that is supported by this exporter.
088         * @return The supported media type
089         */
090        String getMediaType();
091    }
