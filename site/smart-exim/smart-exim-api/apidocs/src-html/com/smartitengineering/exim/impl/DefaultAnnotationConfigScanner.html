053    /**
054     * This registrar is responsible for scanning and containing configuration of
055     * all resources. Whenever a class is looked up whose package has not been
056     * scanned yet, registrar will scanClassForConfig its package to generate cofiguration and
057     * while generating it will keep scanning until it reaches the leaves, i.e. in
058     * this case association to an object that isn't {@link ResourceDomain}
059     * @author imyousuf
060     * @since 0.4
061     */
062    public class DefaultAnnotationConfigScanner
063        implements ClassConfigScanner,
064                   PackageConfigScanner {

068        private static final String HAS_PREFIX = "has";

133        /**
134         * Retrieves the configuration for the resource class, if its not already
135         * generated then it will search the package of the class and generate its
136         * configuraiton.
137         * @param resourceClass The class to scanClassForConfig retrieve configuration for
138         * @return Configuration for the resource; NULL if no resource is available
139         *          or generateable
140         * @throws IllegalArgumentException If resource class is null!
141         */
142        public synchronized EximResourceConfig getResourceConfigForClass(
143            final Class resourceClass) {
144            if (resourceClass == null) {
145                throw new IllegalArgumentException("Resource class can't be null!");
146            }
147            if (configuraitons.containsKey(resourceClass)) {
148                return configuraitons.get(resourceClass);
149            }
150            else {
151                String packageName = resourceClass.getPackage().getName();
152                EximResourceConfig resourceConfig = null;
153                if (!scannedPackages.contains(packageName)) {
154                    resourceConfig = scanPackage(Package.getPackage(packageName),
155                                                 resourceClass);
156                }
157                if (resourceConfig != null) {
158                    return resourceConfig;
159                }
160                else {
161                    //Check direct ancestor
162                    Class parentClass = resourceClass.getSuperclass();
163                    final EximResourceConfig parentClassConfig;
164                    if (parentClass == null || parentClass.equals(Object.class)) {
165                        parentClassConfig = null;
166                    }
167                    else {
168                        parentClassConfig = getResourceConfigForClass(parentClass);
169                    }
170                    if (parentClassConfig == null) {
171                        //Check for directly implemented interfaces
172                        Class[] interfaces = resourceClass.getInterfaces();
173                        Set<Class> interfaceSet = new TreeSet<Class>(new Comparator<Class>() {
174    
175                            public int compare(Class clazz1,
176                                               Class clazz2) {
177                                EximResourceConfig config1 =
178                                                   getConfigurations().get(clazz1);
179                                EximResourceConfig config2 =
180                                                   getConfigurations().get(clazz2);
181                                Integer priority1 = config1.getPriority();
182                                Integer priority2 = config2.getPriority();
183                                if (priority1.equals(priority2)) {
184                                    return clazz1.getName().compareTo(
185                                        clazz2.getName()) * -1;
186                                }
187                                else {
188                                    return priority1.compareTo(priority2) * -1;
189                                }
190                            }
191                        });
192                        for (Class interfaceImpl : interfaces) {
193                            EximResourceConfig interfaceConfig = getResourceConfigForClass(
194                                interfaceImpl);
195                            if (interfaceConfig != null) {
196                                interfaceSet.add(
197                                    interfaceConfig.getDomainClass());
198                            }
199                        }
200                        if (interfaceSet.isEmpty()) {
201                            return null;
202                        }
203                        else {
204                            return getConfigurations().get(interfaceSet.iterator().
205                                next());
206                        }
207                    }
208                    else {
209                        return parentClassConfig;
210                    }
211                }
212            }
213        }

217         * available for future use.

229        /**
230         * Given a method name it returns the property name for it. Currently it
231         * supports methods starting with "get", "is" and "has".
232         * @param methodName The method name from which to derive the property name
233         * @return Property name represented by the read accessor
234         * @throws IllegalArgumentException If prefix is not supported
235         * @throws NullPointerException If method name is null
236         */
237        protected String getPropertyNameFromMethodName(final String methodName)
238            throws IllegalArgumentException,
239                   NullPointerException {
240            StringBuilder propertyNameBuilder =
241                          new StringBuilder(methodName);
242            int cutLength = -1;
243            if (methodName.startsWith(GETTER_PREFIX)) {
244                cutLength = GETTER_PREFIX.length();
245            }
246            else if (methodName.startsWith(IS_PREFIX)) {
247                cutLength = IS_PREFIX.length();
248            }
249            else if (methodName.startsWith(HAS_PREFIX)) {
250                cutLength = HAS_PREFIX.length();
251            }
252            else {
253                throw new IllegalArgumentException(
254                    "Not a valid property read accoessor");
255            }
256            propertyNameBuilder.delete(0, cutLength);
257            char firstChar = propertyNameBuilder.charAt(0);
258            propertyNameBuilder.delete(0, 1);
259            propertyNameBuilder.insert(0, Character.toLowerCase(firstChar));
260            return propertyNameBuilder.toString();
261        }

263        /**
264         * Scan among the class's annotations to find the required configuration for
265         * exporting and importing resources.
266         * @param probableResourceClass The probable resource domain class.
267         * @return The configuration of the class, null if not annotated with 
268         *          {@link ResourceDomain}
269         */
270        protected EximResourceConfig scanClassForConfig(
271            final Class probableResourceClass) {
272            if (probableResourceClass == null) {
273                return null;
274            }
275            if (scannedClasses.contains(probableResourceClass)) {
276                return getConfigurations().get(probableResourceClass);
277            }
278            Annotation annotation = probableResourceClass.getAnnotation(
279                ResourceDomain.class);
280            if (annotation == null) {
281                return null;
282            }
283            EximResourceConfigImpl resourceConfig = new EximResourceConfigImpl();
284            resourceConfig.setDomainClass(probableResourceClass);
285            Name nameAnnotation = (Name) probableResourceClass.getAnnotation(
286                Name.class);
287            if (nameAnnotation != null) {
288                resourceConfig.setName(nameAnnotation.value());
289            }
290            else {
291                resourceConfig.setName(probableResourceClass.getName());
292            }
293            ResourceDomain domainAnnotation = (ResourceDomain) annotation;
294            resourceConfig.setAccessByPropertyEnabled(domainAnnotation.
295                accessByProperty());
296            resourceConfig.setAssociateExportPolicyAsUri(domainAnnotation.
297                exportAsURIByDefault());
298            resourceConfig.setPathToResource(domainAnnotation.path());
299            resourceConfig.setExporterImplemented(DomainSelfExporter.class.
300                isAssignableFrom(probableResourceClass));
301            resourceConfig.setImporterImplemented(DomainSelfImporter.class.
302                isAssignableFrom(probableResourceClass));
303            resourceConfig.setIdentityCustomizerImplemented(
304                IdentityCustomizer.class.isAssignableFrom(probableResourceClass));
305            resourceConfig.setPriority(domainAnnotation.priority());
306                                    resourceConfig.setExportBasicTypesInTypeElementEnabled(domainAnnotation.
307                                                                    exportBasicTypesInTypeElementEnabled());
308            scanMembers(resourceConfig, probableResourceClass);
309            scannedClasses.add(probableResourceClass);
310            //If domain id is not specified then its not a valid domain
311            if (!resourceConfig.isIdentityCustomizerImplemented() &&
312                (resourceConfig.getIdPropertyName() == null || resourceConfig.
313                getIdPropertyName().equals(""))) {
314                return null;
315            }
316            else {
317                configuraitons.put(probableResourceClass, resourceConfig);
318                return resourceConfig;
319            }
320        }

322        /**
323         * Scan a package for extracting configurations of resource domains.
324         * @param packageToScan Package to scanClassForConfig.
325         * @param resourceClass Main class scanClassForConfig requested for
326         * @return The configuration of the resource class, Null if the class is not
327         *          a domain class or resourceClass is null
328         * @throws java.lang.IllegalArgumentException If package is null
329         */
330        protected EximResourceConfig scanPackage(final Package packageToScan,
331                                                 final Class resourceClass)
332            throws IllegalArgumentException {
333            if (packageToScan == null) {
334                throw new IllegalArgumentException();
335            }
336            EximResourceConfig resourceConfig = null;
337            classScanner.scan(new String[]{packageToScan.getName()},
338                              new ClassAnnotationVisitorImpl(callbackHandler,
339                                                             IOFactory.
340                getAnnotationNameForVisitor(ResourceDomain.class)));
341            Set<String> classPaths = resourceVisitCallback.getProbableResources();
342            if (!classPaths.isEmpty()) {
343                for (String classPath : classPaths) {
344                    try {
345                        Class probableResourceClass =
346                              IOFactory.getClassFromVisitorName(classPath);
347                        EximResourceConfig config = scanClassForConfig(
348                            probableResourceClass);
349                        if (config != null && resourceClass != null &&
350                            probableResourceClass.equals(resourceClass)) {
351                            resourceConfig = config;
352                        }
353                    }
354                    catch (ClassNotFoundException ex) {
355                    }
356                    catch (IndexOutOfBoundsException ex) {
357                    }
358                    catch (RuntimeException ex) {
359                    }
360                }
361            }
362            scannedPackages.add(packageToScan.getName());
363            return resourceConfig;
364        }

366        /**
367         * It will scan all member attributes and behavior based on configuration on
368         * the class. It will also scan all inherited attributes and behavior.
369         * @param resourceConfig The config representing the domain class
370         * @param resourceClass The domain class
371         */
372        protected void scanMembers(final EximResourceConfigImpl resourceConfig,
373                                   final Class resourceClass) {
374            if (resourceConfig.isAccessByPropertyEnabled()) {
375                scanMethods(resourceConfig, resourceClass);
376            }
377            else {
378                scanFields(resourceConfig, resourceClass);
379            }
380        }
381    
382        /**
383         * Scans getter methods for discovering associations of the domain and thier
384         * respective configurations. It will only scan public getter methods.
385         * @param resourceConfig The config of the domain resource
386         * @param resourceClass The domain class
387         */
388        protected void scanMethods(final EximResourceConfigImpl resourceConfig,
389                                   final Class resourceClass) {
390            Method[] methods = resourceClass.getMethods();
391            if (methods == null || methods.length <= 0) {
392                return;
393            }
394            for (Method method : methods) {
395                String methodName = method.getName();
396                //Only scan getter methods as of bean spec, that getter methods with
397                //no paratmeters and non-void return types and non-static
398                if (((methodName.startsWith(GETTER_PREFIX) && methodName.length() >
399                    GETTER_PREFIX.length()) || (methodName.startsWith(IS_PREFIX) &&
400                    methodName.length() > IS_PREFIX.length()) || (methodName.
401                    startsWith(HAS_PREFIX) && methodName.length() > HAS_PREFIX.
402                    length())) && method.getReturnType() != null && !method.
403                    getReturnType().equals(Void.class) &&
404                    (method.getParameterTypes() == null ||
405                    method.getParameterTypes().length <= 0) &&
406                    (method.getModifiers() & Modifier.STATIC) <= 0) {
407                    scanGetterMethod(resourceConfig, method);
408                }
409            }
410        }
411    
412        /**
413         * Scans a getter method for annotations which is used to cofigure the
414         * nature of the export
415         * @param resourceConfig The config to populate with configurations
416         * @param method The getter method to scan
417         * @throws java.lang.IllegalArgumentException If its not a getter method
418         *                                            with non-void return type and
419         *                                            with a non-zero length bean
420         *                                            name length and with no
421         *                                            parameter
422         */
423        protected void scanGetterMethod(final EximResourceConfigImpl resourceConfig,
424                                        final Method method)
425            throws IllegalArgumentException {
426            String methodName = method.getName();
427            //Ignore the getClass bean
428            if (method.getName().equals("getClass") || method.getName().equals(
429                "hashCode")) {
430                return;
431            }
432            if (!(((methodName.startsWith(GETTER_PREFIX) && methodName.length() >
433                GETTER_PREFIX.length()) || (methodName.startsWith(IS_PREFIX) &&
434                methodName.length() > IS_PREFIX.length()) || (methodName.startsWith(
435                HAS_PREFIX) && methodName.length() > HAS_PREFIX.length())) &&
436                method.getReturnType() != null && !method.getReturnType().equals(
437                Void.class) && (method.getParameterTypes() == null || method.
438                getParameterTypes().length <= 0) && (method.getModifiers() &
439                Modifier.STATIC) <= 0)) {
440                throw new IllegalArgumentException();
441            }
442            String propertyName = getPropertyNameFromMethodName(methodName);
443            Class returnType = method.getReturnType();
444            scanAnnotatedElement(method, methodName, propertyName, returnType,
445                                 resourceConfig);
446        }

448        /**
449         * Scans fields for discovering associations of the domain and thier
450         * respective configurations
451         * @param resourceConfig The config of the domain resource
452         * @param resourceClass The domain class
453         */
454        protected void scanFields(EximResourceConfigImpl resourceConfig,
455                                  Class resourceClass) {
456            Field[] fields = resourceClass.getDeclaredFields();
457            for (Field field : fields) {
458                scanField(resourceConfig, field);
459            }
460            Class parentClass = resourceClass.getSuperclass();
461            if (!parentClass.equals(Object.class)) {
462                scanFields(resourceConfig, parentClass);
463            }
464        }
465    
466        /**
467         * A field is scanned for gathering configurations for export-import.
468         * @param resourceConfig Configuraton for the field association
469         * @param field The field to scan
470         */
471        protected void scanField(final EximResourceConfigImpl resourceConfig,
472                                 final Field field) {
473            if ((field.getModifiers() & Modifier.STATIC) > 0) {
474                return;
475            }
476            String propertyName = field.getName();
477            Class propertyType = field.getType();
478            scanAnnotatedElement(field, propertyName, propertyName, propertyType,
479                                 resourceConfig);
480        }

482        /**
483         * Scan an ennotated element to extract configuration information
484         * @param element Element to scan
485         * @param propertyName The name of the property scanning
486         * @param accessorName The name of the property accessor
487         * @param propertyType The type of the property
488         * @param resourceConfig The configuration for the domain class
489         * @throws java.lang.IllegalArgumentException If any argument is null
490         */
491        protected void scanAnnotatedElement(final AnnotatedElement element,
492                                            final String accessorName,
493                                            final String propertyName,
494                                            final Class propertyType,
495                                            final EximResourceConfigImpl resourceConfig)
496            throws IllegalArgumentException {
497            if (element == null || propertyName == null || propertyType == null ||
498                resourceConfig == null) {
499                throw new IllegalArgumentException();
500            }
501            AssociationConfigImpl configImpl =
502                                  new AssociationConfigImpl();
503            configImpl.setAccessorName(accessorName);
504            Name nameAnnotation = element.getAnnotation(Name.class);
505            if (nameAnnotation != null) {
506                configImpl.setName(nameAnnotation.value());
507            }
508            else {
509                configImpl.setName(propertyName);
510            }
511            configImpl.setAssociationType(AssociationConfig.AssociationType.
512                getAssociationType(propertyType));
513            Eager eager = element.getAnnotation(Eager.class);
514            configImpl.setStringProviderImplemented(StringValueProvider.class.
515                isAssignableFrom(propertyType));
516            configImpl.setEagerSet(eager != null);
517            Export annotation = element.getAnnotation(Export.class);
518            if (annotation != null) {
519                configImpl.setItToBeExportedAsUri(!annotation.asObject());
520                configImpl.setTransient(annotation.isTransient());
521            }
522            else {
523                configImpl.setItToBeExportedAsUri(false);
524                configImpl.setTransient(false);
525            }
526            resourceConfig.getAssociationConfigs().put(propertyName, configImpl);
527            Id id = element.getAnnotation(Id.class);
528            if (id != null) {
529                resourceConfig.setIdPropertyName(propertyName);
530                resourceConfig.setIdPrefix(id.path());
531            }
532        }

534        /**
535         * The visitor callback to be notified for dmain classes.
536         */
537        protected static class ResourceVisitCallback
538            implements VisitCallback<AnnotationConfig> {
539    
540            private Set<String> probableResources = new HashSet<String>();
541    
542            /**
543             * When a requested annotation is been parsed the callback will be
544             * triggered and it will maintain a {@link Set} of scanned classes and
545             * mark them as probable resource
546             * @param config The annotation been parsed
547             */
548            public void handle(AnnotationConfig config) {
549                probableResources.add(config.getClassName());
550            }
551    
552            /**
553             * Return the probable resources scanned upto now
554             * @return Probable resources
555             */
556            public Set<String> getProbableResources() {
557                return probableResources;
558            }
559        }
