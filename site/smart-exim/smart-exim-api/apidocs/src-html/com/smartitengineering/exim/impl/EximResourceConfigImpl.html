019    package com.smartitengineering.exim.impl;
020    
021    import com.smartitengineering.exim.AssociationConfig;
022    import com.smartitengineering.exim.EximResourceConfig;
023    import java.util.HashMap;
024    import java.util.Map;
025    
026    /**
027     * This implementation represents the configuration of a class. 2 configuration
028     * objects are considered to be equal if they represent the same class.
029     * @author imyousuf
030     * @since 0.4
031     */
032    public class EximResourceConfigImpl
033                                    implements EximResourceConfig {
034    
035                    private boolean associateExportPolicyAsUri;
036                    private boolean accessByPropertyEnabled;
037                    private boolean identityCustomizerImplemented;
038                    private boolean exporterImplemented;
039                    private boolean importerImplemented;
040                    private boolean exportBasicTypesInTypeElementEnabled;
041                    private Class domainClass;
042                    private String pathToResource;
043                    private String idPrefix;
044                    private String idPropertyName;
045                    private String name;
046                    private int priority;
047                    private Map<String, AssociationConfig> associationConfigs;
048    
049                    public EximResourceConfigImpl() {
050                                    associationConfigs = new HashMap<String, AssociationConfig>();
051                    }
052    
053                    public void setAccessByPropertyEnabled(boolean accessByPropertyEnabled) {
054                                    this.accessByPropertyEnabled = accessByPropertyEnabled;
055                    }
056    
057                    public void setAssociateExportPolicyAsUri(boolean associateExportPolicyAsUri) {
058                                    this.associateExportPolicyAsUri = associateExportPolicyAsUri;
059                    }
060    
061                    public void setAssociationConfigs(
062                                                    Map<String, AssociationConfig> associationConfigs) {
063                                    if (associationConfigs != null) {
064                                                    this.associationConfigs.clear();
065                                                    this.associationConfigs.putAll(associationConfigs);
066                                    }
067                    }
068    
069                    public void setDomainClass(Class domainClass) {
070                                    this.domainClass = domainClass;
071                    }
072    
073                    public void setExporterImplemented(boolean exporterImplemented) {
074                                    this.exporterImplemented = exporterImplemented;
075                    }
076    
077                    public void setIdPropertyName(String idPropertyName) {
078                                    this.idPropertyName = idPropertyName;
079                    }
080    
081                    public void setIdPrefix(String idPrefix) {
082                                    this.idPrefix = idPrefix;
083                    }
084    
085                    public void setIdentityCustomizerImplemented(
086                                                    boolean identityCustomizerImplemented) {
087                                    this.identityCustomizerImplemented = identityCustomizerImplemented;
088                    }
089    
090                    public void setImporterImplemented(boolean importerImplemented) {
091                                    this.importerImplemented = importerImplemented;
092                    }
093    
094                    public void setPathToResource(String pathToResource) {
095                                    this.pathToResource = pathToResource;
096                    }
097    
098                    public void setName(String name) {
099                                    this.name = name;
100                    }
101    
102                    public void setPriority(int priority) {
103                                    this.priority = priority;
104                    }
105    
106                    public void setExportBasicTypesInTypeElementEnabled(
107                                                    boolean exportBasicTypesInTypeElementEnabled) {
108                                    this.exportBasicTypesInTypeElementEnabled =
109                                    exportBasicTypesInTypeElementEnabled;
110                    }
111    
112                    public boolean isAccessByPropertyEnabled() {
113                                    return accessByPropertyEnabled;
114                    }
115    
116                    public boolean isAssociateExportPolicyAsUri() {
117                                    return associateExportPolicyAsUri;
118                    }
119    
120                    public Map<String, AssociationConfig> getAssociationConfigs() {
121                                    return associationConfigs;
122                    }
123    
124                    public Class getDomainClass() {
125                                    return domainClass;
126                    }
127    
128                    public boolean isExporterImplemented() {
129                                    return exporterImplemented;
130                    }
131    
132                    public String getIdPropertyName() {
133                                    return idPropertyName;
134                    }
135    
136                    public String getIdPrefix() {
137                                    return idPrefix;
138                    }
139    
140                    public boolean isIdentityCustomizerImplemented() {
141                                    return identityCustomizerImplemented;
142                    }
143    
144                    public boolean isImporterImplemented() {
145                                    return importerImplemented;
146                    }
147    
148                    public String getPathToResource() {
149                                    return pathToResource;
150                    }
151    
152                    public String getName() {
153                                    return name;
154                    }
155    
156                    public int getPriority() {
157                                    return priority;
158                    }
159    
160                    public boolean isExportBasicTypesInTypeElementEnabled() {
161                                    return exportBasicTypesInTypeElementEnabled;
162                    }
163    
164                    @Override
165                    public String toString() {
166                                    StringBuilder toStringBuilder = new StringBuilder();
167                                    toStringBuilder.append(super.toString());
168                                    toStringBuilder.append("\n");
169                                    toStringBuilder.append("\nassociateExportPolicyAsUri: ");
170                                    toStringBuilder.append(associateExportPolicyAsUri);
171                                    toStringBuilder.append("\naccessByPropertyEnabled: ");
172                                    toStringBuilder.append(accessByPropertyEnabled);
173                                    toStringBuilder.append("\nidentityCustomizerImplemented: ");
174                                    toStringBuilder.append(identityCustomizerImplemented);
175                                    toStringBuilder.append("\nexporterImplemented: ");
176                                    toStringBuilder.append(exporterImplemented);
177                                    toStringBuilder.append("\nimporterImplemented: ");
178                                    toStringBuilder.append(importerImplemented);
179                                    toStringBuilder.append("\ndomainClass: ");
180                                    toStringBuilder.append(domainClass);
181                                    toStringBuilder.append("\npathToResource: ");
182                                    toStringBuilder.append(pathToResource);
183                                    toStringBuilder.append("\nidPrefix: ");
184                                    toStringBuilder.append(idPrefix);
185                                    toStringBuilder.append("\nidPropertyName: ");
186                                    toStringBuilder.append(idPropertyName);
187                                    toStringBuilder.append("\nname: ");
188                                    toStringBuilder.append(name);
189                                    toStringBuilder.append("\npriority: ");
190                                    toStringBuilder.append(priority);
191                                    toStringBuilder.append("\nexportBasicTypesInTypeElementEnabled: ");
192                                    toStringBuilder.append(exportBasicTypesInTypeElementEnabled);
193                                    toStringBuilder.append("\nassociationConfigs: ");
194                                    for (Map.Entry<String, AssociationConfig> configEntry :
195                                                     associationConfigs.entrySet()) {
196                                                    toStringBuilder.append("\n\t");
197                                                    toStringBuilder.append(configEntry.getKey());
198                                                    toStringBuilder.append(": ");
199                                                    toStringBuilder.append(configEntry.getValue().toString().replaceAll(
200                                                                                    "\n", "\n\t\t"));
201                                    }
202                                    return toStringBuilder.toString();
203                    }
204    }
