019    package com.smartitengineering.exim.impl;
020    
021    import com.smartitengineering.exim.AssociationConfig;
022    import com.smartitengineering.exim.EximResourceConfig;
023    import java.util.Map;
024    
025    /**
026     * This implementation represents the configuration of a class. 2 configuration
027     * objects are considered to be equal if they represent the same class.
028     * @author imyousuf
029     * @since 0.4
030     */
031    public class EximResourceConfigImpl
032        implements EximResourceConfig {
033    
034        private boolean associateExportPolicyAsUri;
035        private boolean accessByPropertyEnabled;
036        private boolean identityCustomizerImplemented;
037        private boolean exporterImplemented;
038        private boolean importerImplemented;
039        private Class domainClass;
040        private String pathToResource;
041        private String idPrefix;
042        private String idPropertyName;
043        private String name;
044        private Map<String, AssociationConfig> associationConfigs;
045    
046        public void setAccessByPropertyEnabled(boolean accessByPropertyEnabled) {
047            this.accessByPropertyEnabled = accessByPropertyEnabled;
048        }
049    
050        public void setAssociateExportPolicyAsUri(boolean associateExportPolicyAsUri) {
051            this.associateExportPolicyAsUri = associateExportPolicyAsUri;
052        }
053    
054        public void setAssociationConfigs(
055            Map<String, AssociationConfig> associationConfigs) {
056            this.associationConfigs = associationConfigs;
057        }
058    
059        public void setDomainClass(Class domainClass) {
060            this.domainClass = domainClass;
061        }
062    
063        public void setExporterImplemented(boolean exporterImplemented) {
064            this.exporterImplemented = exporterImplemented;
065        }
066    
067        public void setIdPropertyName(String idPropertyName) {
068            this.idPropertyName = idPropertyName;
069        }
070    
071        public void setIdPrefix(String idPrefix) {
072            this.idPrefix = idPrefix;
073        }
074    
075        public void setIdentityCustomizerImplemented(
076            boolean identityCustomizerImplemented) {
077            this.identityCustomizerImplemented = identityCustomizerImplemented;
078        }
079    
080        public void setImporterImplemented(boolean importerImplemented) {
081            this.importerImplemented = importerImplemented;
082        }
083    
084        public void setPathToResource(String pathToResource) {
085            this.pathToResource = pathToResource;
086        }
087    
088        public void setName(String name) {
089            this.name = name;
090        }
091    
092        public boolean isAccessByPropertyEnabled() {
093            return accessByPropertyEnabled;
094        }
095    
096        public boolean isAssociateExportPolicyAsUri() {
097            return associateExportPolicyAsUri;
098        }
099    
100        public Map<String, AssociationConfig> getAssociationConfigs() {
101            return associationConfigs;
102        }
103    
104        public Class getDomainClass() {
105            return domainClass;
106        }
107    
108        public boolean isExporterImplemented() {
109            return exporterImplemented;
110        }
111    
112        public String getIdPropertyName() {
113            return idPropertyName;
114        }
115    
116        public String getIdPrefix() {
117            return idPrefix;
118        }
119    
120        public boolean isIdentityCustomizerImplemented() {
121            return identityCustomizerImplemented;
122        }
123    
124        public boolean isImporterImplemented() {
125            return importerImplemented;
126        }
127    
128        public String getPathToResource() {
129            return pathToResource;
130        }
131    
132        public String getName() {
133            return name;
134        }
135    }
