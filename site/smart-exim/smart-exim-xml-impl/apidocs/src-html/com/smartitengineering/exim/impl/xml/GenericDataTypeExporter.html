019    package com.smartitengineering.exim.impl.xml;
020    
021    import com.smartitengineering.exim.AssociationConfig.AssociationType;
022    import java.io.IOException;
023    import java.util.Collections;
024    import java.util.Date;
025    import javax.xml.transform.sax.TransformerHandler;
026    import org.apache.commons.lang.StringUtils;
027    import org.apache.commons.lang.time.DateFormatUtils;
028    import org.xml.sax.helpers.AttributesImpl;
029    
030    /**
031     * Exporter of generic data types;
032     * @author imyousuf
033     * @since 0.4
034     */
035    public class GenericDataTypeExporter
036                                    extends AbstractElementExporter
037                                    implements ElementExporter {
038    
039                    private static ElementExporter exporter;
040    
041                    static {
042                                    exporter = new GenericCollectionElementExporter();
043                                    for (AssociationType type : exporter.getSupportedTypes()) {
044                                                    XMLExporterImpl.addHandler(type, exporter);
045                                    }
046                    }
047    
048                    public static ElementExporter getInstance() {
049                                    return exporter;
050                    }
051    
052                    public GenericDataTypeExporter() {
053                                    Collections.addAll(types, AssociationType.TYPE_DATE,
054                                                                    AssociationType.TYPE_STRING, AssociationType.TYPE_DOUBLE,
055                                                                    AssociationType.TYPE_FLOAT, AssociationType.TYPE_INTEGER,
056                                                                    AssociationType.TYPE_LONG, AssociationType.TYPE_BOOLEAN);
057                    }
058    
059                    public void exportElement(AssociationType type,
060                                                                                                                            Object object,
061                                                                                                                            TransformerHandler handler)
062                                                    throws IOException {
063                                    if (type == null || object == null || handler == null) {
064                                                    throw new IOException("All parameters are required!");
065                                    }
066                                    final String elementName = type.getSimpleName();
067                                    final String value;
068                                    switch (type) {
069                                                    case TYPE_DATE:
070                                                                    value = DateFormatUtils.ISO_DATETIME_TIME_ZONE_FORMAT.format(
071                                                                                                    (Date) object);
072                                                                    break;
073                                                    case TYPE_STRING:
074                                                    case TYPE_INTEGER:
075                                                    case TYPE_LONG:
076                                                    case TYPE_FLOAT:
077                                                    case TYPE_DOUBLE:
078                                                    case TYPE_BOOLEAN:
079                                                    default:
080                                                                    value = object == null ? null : object.toString();
081                                                                    break;
082                                    }
083                                    try {
084                                                    AttributesImpl atts = new AttributesImpl();
085                                                    handler.startElement(EXIM_COLLECN_URI, EXIM_COLLECTION_NS,
086                                                                                    elementName,
087                                                                                    atts);
088                                                    if (AssociationType.TYPE_STRING.equals(type)) {
089                                                                    handler.startCDATA();
090                                                    }
091                                                    final char[] chars;
092                                                    if (StringUtils.isBlank(value)) {
093                                                                    chars = "".toCharArray();
094                                                    }
095                                                    else {
096                                                                    chars = value.toCharArray();
097                                                    }
098                                                    handler.characters(chars, 0, chars.length);
099                                                    if (AssociationType.TYPE_STRING.equals(type)) {
100                                                                    handler.endCDATA();
101                                                    }
102                                                    handler.endElement(EXIM_COLLECN_URI, EXIM_COLLECTION_NS, elementName);
103                                    }
104                                    catch (Exception exception) {
105                                                    throw new IOException(exception);
106                                    }
107                    }
108    }
