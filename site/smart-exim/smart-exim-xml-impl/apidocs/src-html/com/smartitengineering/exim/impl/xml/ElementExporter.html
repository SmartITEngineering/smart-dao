019    package com.smartitengineering.exim.impl.xml;
020    
021    import com.smartitengineering.exim.AssociationConfig.AssociationType;
022    import java.io.IOException;
023    import java.util.Collection;
024    import javax.xml.transform.sax.TransformerHandler;
025    
026    /**
027     * An interface for different types of association exporter implementation. All
028     * its implementation should be designed in singleton thread safe way.
029     * @author imyousuf
030     * @since 0.4
031     */
032    public interface ElementExporter {
033    
034                    public static final String EXIM_COLLECTION_NS = "exim-collection";
035                    public static final String EXIM_COLLECN_URI =
036                                                                                                                             "http://www.smartitengineering.com/smart-dao/exim/collections";
037                    public static final String EXIM_BASIC_TYPES_NS = "exim-basic-types";
038                    public static final String EXIM_BASIC_TYPES_URI =
039                                                                                                                             "http://www.smartitengineering.com/smart-dao/exim/basic-types";
040    
041                    /**
042                     * This operation is resposnsible for exporting XML elements of type it
043                     * supports. It will use other type exporters for exporting and will only be
044                     * responsible for the types it supports.
045                     * @param type Type to export
046                     * @param object Object to export
047                     * @param handler Handler to use to create elements and attributes
048                     * @throws IOException If any error while exporting
049                     */
050                    public void exportElement(final AssociationType type,
051                                                                                                                            final Object object,
052                                                                                                                            final TransformerHandler handler)
053                                                    throws IOException;
054    
055                    public Collection<AssociationType> getSupportedTypes();
056    }
