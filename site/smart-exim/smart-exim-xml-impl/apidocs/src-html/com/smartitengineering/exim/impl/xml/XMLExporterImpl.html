033    import java.lang.reflect.Type;

035    import java.util.List;

037    import javax.xml.transform.OutputKeys;
038    import javax.xml.transform.Transformer;
039    import javax.xml.transform.TransformerConfigurationException;
040    import javax.xml.transform.sax.SAXTransformerFactory;
041    import javax.xml.transform.sax.TransformerHandler;
042    import javax.xml.transform.stream.StreamResult;
043    import org.xml.sax.SAXException;
044    
045    /**
046     *
047     * @author imyousuf
048     * @since 0.4
049     */
050    public class XMLExporterImpl
051                                    implements Exporter {
052    
053                    private String encoding;
054                    private String indent;
055                    private String method;
056                    private static Map<AssociationType, ElementExporter> handlers;
057    
058                    static {
059                                    handlers = new HashMap<AssociationType, ElementExporter>(
060                                                                    AssociationType.values().length);
061                                    String packageName = ConfigRegistrar.class.getPackage().getName();
062                                    ClassScanner scanner = IOFactory.getDefaultClassScanner();
063                                    scanner.scan(new String[] {packageName},
064                                                                    new ClassInstanceVisitorImpl(IOFactory.getClassNameForVisitor(
065                                                                    ElementExporter.class), new VisitCallback<Config>() {
066    
067                                                    public void handle(Config config) {
068                                                                    try {
069                                                                                    IOFactory.getClassFromVisitorName(config.getClassName());
070                                                                    }
071                                                                    catch (Exception ex) {
072                                                                                    ex.printStackTrace();
073                                                                    }
074                                                    }
075                                    }));
076                    }
077    
078                    static void addHandler(AssociationType type,
079                                                                                                             ElementExporter exporter) {
080                                    handlers.put(type, exporter);
081                    }
082    
083                    static ElementExporter getElementExporter(AssociationType type) {
084                                    return handlers.get(type);
085                    }
086    
087                    public XMLExporterImpl() {
088                                    encoding = "UTF-8";
089                                    indent = "yes";
090                                    method = "xml";
091                    }
092    
093                    public boolean isExportSupported(Class clazz,
094                                                                                                                                                     Type genericType,
095                                                                                                                                                     Annotation[] annotations) {
096                                    if (clazz == null) {
097                                                    return false;
098                                    }
099                                    AssociationType type = AssociationType.getAssociationType(clazz);
100                                    if (type != null && type != AssociationType.TYPE_OBJECT &&
101                                                    type != AssociationType.TYPE_ARRAY) {
102                                                    return true;
103                                    }
104                                    EximResourceConfig config = ConfigRegistrar.getConfigForClass(
105                                                                    getComponentClass(clazz));
106                                    if (config != null) {
107                                                    return true;
108                                    }
109                                    return false;
110                    }
111    
112                    public long getAfterExportLength(Object object,
113                                                                                                                                                     Class clazz,
114                                                                                                                                                     Type genericType,
115                                                                                                                                                     Annotation[] annotations) {
116                                    return -1;
117                    }
118    
119                    public void exportObject(Object object,
120                                                                                                                     Class clazz,
121                                                                                                                     Type genericType,
122                                                                                                                     Annotation[] annotations,
123                                                                                                                     OutputStream outputStream,
124                                                                                                                     Map<Object, List<Object>> headers)
125                                                    throws IOException {
126                                    if (outputStream == null || clazz == null || object == null) {
127                                                    throw new IOException(
128                                                                                    "OutputStream or Object or Class can't be null!");
129                                    }
130                                    try {
131                                                    SAXTransformerFactory factory =
132                                                                                                                                            (SAXTransformerFactory) SAXTransformerFactory.
133                                                                                    newInstance();
134                                                    TransformerHandler handler = factory.newTransformerHandler();
135                                                    StreamResult result = new StreamResult(outputStream);
136                                                    Transformer transformer = handler.getTransformer();
137                                                    transformer.setOutputProperty(OutputKeys.ENCODING, encoding);
138                                                    transformer.setOutputProperty(OutputKeys.INDENT, indent);
139                                                    transformer.setOutputProperty(OutputKeys.METHOD, method);
140                                                    handler.setResult(result);
141                                                    handler.startDocument();
142                                                    try {
143                                                                    AssociationType type = AssociationType.getAssociationType(clazz);
144                                                                    ElementExporter exporter = getElementExporter(type);
145                                                                    if (exporter == null) {
146                                                                                    throw new UnsupportedOperationException("Type not supported yet! - " +
147                                                                                                                                                                                                                                                    type);
148                                                                    }
149                                                                    exporter.exportElement(type, object, handler);
150    
151                                                    }
152                                                    finally {
153                                                                    handler.endDocument();
154                                                    }
155                                    }
156                                    catch (SAXException ex) {
157                                                    throw new IOException(ex);
158                                    }
159                                    catch (TransformerConfigurationException ex) {
160                                                    throw new IOException(ex);
161                                    }
162                    }
163    
164                    public String getMediaType() {
165                                    return "application/xml";
166                    }
167    
168                    public String getEncoding() {
169                                    return encoding;
170                    }
171    
172                    public void setEncoding(String encoding) {
173                                    this.encoding = encoding;
174                    }
175    
176                    public String getIndent() {
177                                    return indent;
178                    }
179    
180                    public void setIndent(String indent) {
181                                    this.indent = indent;
182                    }
183    
184                    public String getMethod() {
185                                    return method;
186                    }
187    
188                    public void setMethod(String method) {
189                                    this.method = method;
190                    }
191    
192                    private Class getComponentClass(Class clazz) {
193                                    if (clazz == null) {
194                                                    throw new IllegalArgumentException();
195                                    }
196                                    return clazz.isArray() ? clazz.getComponentType() : clazz;
197                    }
