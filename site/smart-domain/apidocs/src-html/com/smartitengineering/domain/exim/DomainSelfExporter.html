019    package com.smartitengineering.domain.exim;
020    
021    import java.io.IOException;
022    import java.io.OutputStream;
023    
024    /**
025     * As the name suggests the API indicates that a domain is able to export itself
026     * and to determine for which MIME types it can it has an operation dedicated op
027     * for it.
028     * @author imyousuf
029     * @since 0.4
030     */
031    public interface DomainSelfExporter<T> {
032    
033        /**
034         * It provides the operation for clients to determine whether the domain is
035         * able to export itself to the MIME type the client wants.
036         * @param mimeType Type to export to
037         * @return True if and only if the mimeType is supported for export
038         */
039        public boolean isMIMESupported(String mimeType);
040    
041        /**
042         * Exports the domain to the specified stream in specified type
043         * @param mimeType Type to export to.
044         * @param outputStream Stream to export to.
045         * @throws IllegalArgumentException If either argument is null or MIME type
046         *                                  isn't supported.
047         * @throws IOException If any error occurrs while writing to stream.
048         */
049        public void exportObject(String mimeType,
050                                 OutputStream outputStream)
051            throws IllegalArgumentException,
052                   IOException;
053    }
