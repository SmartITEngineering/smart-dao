041        private static final String DEFAULT_PATH = "target/jgit-impl/.git";
042        private JGitImpl jGitImpl;
043        private boolean finished = false;
044        private static String firstRevisionId = "";

046        private boolean isFinished() {
047            return finished;
048        }

050        public JGitImplTest(String testName) {
051            super(testName);
052        }

054        @Override
055        protected void setUp()
056            throws Exception {
057            super.setUp();
058            jGitImpl = new JGitImpl();

060            jGitImpl.init();
061            finished = false;
062        }

064        @Override
065        protected void tearDown()
066            throws Exception {
067            super.tearDown();

071        public void testStore() {
072            final Revision revision =
073                VersionAPI.createRevision(VersionAPI.createResource("a/a.xml",
074                "Content of a/a"), null);
075    
076            jGitImpl.store(VersionAPI.createCommit(Arrays.asList(revision,VersionAPI.
077                createRevision(
078                VersionAPI.createResource("b/a.xml", "Content of b/a"),
079                null)), null,
080                null, "Commit message for A", VersionAPI.createAuthor(
081                "Imran M Yousuf", "imran@smartitengineering.com"), null), new WriterCallback() {
082    
083                public void handle(Commit commit,
084                                   WriteStatus status,
085                                   String comment,
086                                   Throwable error) {
087                    if (error != null) {
088                        error.printStackTrace();
089                    }
090                    assertEquals(WriteStatus.STORE_PASS, status);
091                    assertNull(error);
092                    assertNotNull(revision.getRevisionId());
093                    firstRevisionId = revision.getRevisionId();
094                    finished = true;
095                }
096            });
097            while (!isFinished()) {
098                try {
099                    Thread.sleep(10);
100                }
101                catch (InterruptedException ex) {
102                    ex.printStackTrace();
103                }
104            }
105            finished = false;
106        }

108        public void testReadResource() {
109            Resource a = jGitImpl.getResource("a/a.xml");
110            assertNotNull(a);
111            assertEquals("a/a.xml", a.getId());
112            assertEquals("Content of a/a", a.getContent());
113            finished = false;
114            jGitImpl.store(VersionAPI.createCommit(Arrays.asList(VersionAPI.
115                createRevision(VersionAPI.createResource("a/a.xml",
116                "UPDATE-1 Content of a/a"),
117                null)), null,
118                null, "Commit message for A-2", VersionAPI.createAuthor(
119                "Imran M Yousuf", "imran@smartitengineering.com"), null), new WriterCallback() {
120    
121                public void handle(Commit commit,
122                                   WriteStatus status,
123                                   String comment,
124                                   Throwable error) {
125                    if (error != null) {
126                        error.printStackTrace();
127                    }
128                    assertEquals(status, WriteStatus.STORE_PASS);
129                    assertNull(error);
130                    finished = true;
131                }
132            });
133            while (!isFinished()) {
134                try {
135                    Thread.sleep(10);
136                }
137                catch (InterruptedException ex) {
138                    ex.printStackTrace();
139                }
140            }
141            finished = false;
142            Resource b = jGitImpl.getResource("b/a.xml");
143            assertEquals("b/a.xml", b.getId());
144            assertEquals("Content of b/a", b.getContent());
145            a = jGitImpl.getResource("a/a.xml");
146            assertEquals("a/a.xml", a.getId());
147            assertEquals("UPDATE-1 Content of a/a", a.getContent());
148            if(StringUtils.isNotBlank(firstRevisionId)) {
149                a = jGitImpl.getResourceByRevision(firstRevisionId, "a/a.xml");
150                assertEquals("a/a.xml", a.getId());
151                assertEquals("Content of a/a", a.getContent());
152            }
153            /**
154             * Try non-existing resource id and revision id
155             */
156            String rubbish = "asdassdasd";
157            try {
158                jGitImpl.getResource(rubbish);
159                fail("Should not be able to return");
160            }
161            catch(RuntimeException ex) {
162            }
163            try {
164                jGitImpl.getResourceByRevision(rubbish, rubbish);
165                fail("Should not be able to return");
166            }
167            catch(RuntimeException ex) {
168            }
169        }

208        public void testVersionedResource() {
209            VersionedResource versionedResource = jGitImpl.getVersionedResource(
210                "/////a/a.xml///");
211            assertNotNull(versionedResource);
212            assertNotNull(versionedResource.getHeadVersionResource());
213            Resource a = versionedResource.getHeadVersionResource();
214            assertEquals("a/a.xml", a.getId());
215            assertEquals("UPDATE-1 Content of a/a", a.getContent());
216            assertEquals(2, versionedResource.getRevisions().size());
217            Iterator<Revision> iterator =
218                versionedResource.getRevisions().iterator();
219            a = iterator.next().getResource();
220            assertEquals("a/a.xml", a.getId());
221            assertEquals("UPDATE-1 Content of a/a", a.getContent());
222            a = iterator.next().getResource();
223            assertEquals("a/a.xml", a.getId());
224            assertEquals("Content of a/a", a.getContent());
225            try {
226                jGitImpl.getVersionedResource("aaaa/a.xml");
227                fail("Should not return!");
228            }
229            catch(RuntimeException exception) {
230            }
231        }
