019    package com.smartitengineering.version.impl.jgit.service;
020    
021    /**
022     *
023     * @author imyousuf
024     */
025    public class RCSConfig {
026    
027        private int concurrentWriteOperations;
028        private boolean allowNoChangeCommit;
029        private String repositoryReadPath;
030        private String repositoryWritePath;
031    
032        public boolean isAllowNoChangeCommit() {
033            return allowNoChangeCommit;
034        }
035    
036        public void setAllowNoChangeCommit(boolean allowNoChangeCommit) {
037            this.allowNoChangeCommit = allowNoChangeCommit;
038        }
039    
040        public int getConcurrentWriteOperations() {
041            return concurrentWriteOperations;
042        }
043    
044        public void setConcurrentWriteOperations(int concurrentWriteOperations) {
045            this.concurrentWriteOperations = concurrentWriteOperations;
046        }
047    
048        public String getRepositoryReadPath() {
049            return repositoryReadPath;
050        }
051    
052        public void setRepositoryReadPath(String repositoryPath) {
053            this.repositoryReadPath = repositoryPath;
054        }
055    
056        public String getRepositoryWritePath() {
057            return repositoryWritePath;
058        }
059    
060        public void setRepositoryWritePath(String repositoryWritePath) {
061            this.repositoryWritePath = repositoryWritePath;
062        }
063    }
