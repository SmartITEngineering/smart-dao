019    package com.smartitengineering.version.impl.jgit;
020    
021    import com.smartitengineering.version.impl.jgit.domain.Author;
022    import com.smartitengineering.version.impl.jgit.domain.Commit;
023    import com.smartitengineering.version.impl.jgit.domain.Resource;
024    import com.smartitengineering.version.impl.jgit.domain.Revision;
025    
026    /**
027     * Represents enum for property name of a query parameter to be used for
028     * querying to the Version API. While using it please pay attention to the types
029     * of each property and use them wisely; we would like to mention that we will
030     * pass through your query params directly without verifying them.
031     * @author imyousuf
032     */
033    public enum SearchProperties
034        implements SearchParam {
035    
036        /**
037         * The java.util.Date of a commit.
038         */
039        COMMIT_DATE(Commit.PROP_DATETIME),
040        /**
041         * The string log message of a commit
042         */
043        COMMIT_MSG(Commit.PROP_COMMITMESSAGE),
044        /**
045         * Use revisions as a nested parameters to filter commits by revisions props
046         * available as in 'REVISION*'
047         */
048        COMMIT_REVISIONS(Commit.PROP_REVISIONS),
049        /**
050         * The full name of an committer, a string.
051         */
052        COMMITTER_NAME(Commit.PROP_COMMITTER + '.' + Author.PROP_NAME),
053        /**
054         * The email address of a committer, a string.
055         */
056        COMMITTER_EMAIL(Commit.PROP_COMMITTER + '.' + Author.PROP_EMAIL),
057        /**
058         * Specify the interested resource id and it accepts a String.
059         */
060        REVISION_RESOURCE(Revision.PROP_RESOURCE + '.' + Resource.PROP_RESOURCEID),
061        /**
062         * Use this BOOLEAN field to specify to search within HEAD revisions only or
063         * not
064         */
065        REVISION_HEAD(Revision.PROP_HEADREVISION),
066        /**
067         * Filter whether the revision's resource is deleted or not, its a boolean.
068         */
069        REVISION_RESOURCE_DELETED(Revision.PROP_RESOURCE + '.' + Resource.PROP_DELETE),
070        /**
071         * <p>Use this to search for properties of a commit of a revision</p>
072         * <p>While others can be used as property directly, in order to use this we
073         * will need to create a nested property param and then use COMMIT* props
074         * perform search.</p>
075         */
076        REVISION_COMMIT(Revision.PROP_COMMIT);
077        private final String propertyName;
078    
079        private SearchProperties(final String propertyName) {
080            this.propertyName = propertyName;
081        }
082    
083        public String getPropertyName() {
084            return propertyName;
085        }
086    }
