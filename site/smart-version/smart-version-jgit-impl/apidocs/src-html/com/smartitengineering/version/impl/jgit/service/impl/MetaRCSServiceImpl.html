125        public String[] getVersionsForResource(final String resourceId) {
126            if (StringUtils.isBlank(resourceId)) {
127                throw new IllegalArgumentException("Blank resource not supported!");
128            }
129            final List<String> revisions =
130                revisionReader.<String>getOtherList(
131                QueryParameterFactory.<String>getStringLikePropertyParam(
132                new StringBuilder(Revision.PROP_RESOURCE).append('.').
133                append(Resource.PROP_RESOURCEID).toString(),
134                resourceId, MatchMode.EXACT),
135                QueryParameterFactory.getPropProjectionParam(
136                Revision.PROP_REVISIONID),
137                QueryParameterFactory.<Boolean>getEqualPropertyParam(
138                SearchProperties.REVISION_RESOURCE_DELETED.getPropertyName(), false),
139                QueryParameterFactory.getOrderByParam("id", Order.DESC));
140            return revisions.toArray(new String[0]);
141        }
142    
143        public String getHeadVersionForResource(final String resourceId) {
144            if (StringUtils.isBlank(resourceId)) {
145                throw new IllegalArgumentException("Blank resource not supported!");
146            }
147            final String revision =
148                (String) revisionReader.getOther(QueryParameterFactory.<String>
149                getStringLikePropertyParam(new StringBuilder(
150                Revision.PROP_RESOURCE).append('.').
151                append(Resource.PROP_RESOURCEID).toString(),
152                resourceId, MatchMode.EXACT),
153                QueryParameterFactory.<Boolean>getEqualPropertyParam(
154                Revision.PROP_HEADREVISION, true),
155                QueryParameterFactory.getPropProjectionParam(
156                Revision.PROP_REVISIONID),
157                QueryParameterFactory.<Boolean>getEqualPropertyParam(
158                SearchProperties.REVISION_RESOURCE_DELETED.getPropertyName(), false));
159            return revision;
160        }

162        public void saveResources(
163            final com.smartitengineering.version.api.Commit commit) {
164            Commit commitDomain = MetaFactory.transformAPICommit(commit);
165            persistCommit(commitDomain);
166        }
167    
168        private void persistCommit(Commit commitDomain)
169            throws IllegalStateException {
170            //Set all revisions in commit to 'head' and update all the revisions for
171            //the resources already in the system to 'not head'
172            if (!commitDomain.isValid()) {
173                throw new IllegalStateException(
174                    "Commit domain's current state is not valid!");
175            }
176            Set<String> resourceIds = new HashSet<String>();
177            Set<Revision> currentHeads = new HashSet<Revision>();
178            for (Revision revision : commitDomain.getRevisions()) {
179                revision.setHeadRevision(true);
180                String resourceId = revision.getResource().getResourceId();
181                resourceIds.add(resourceId);
182            }
183            currentHeads = new HashSet<Revision>(revisionReader.getList(
184                QueryParameterFactory.<String>getIsInPropertyParam(new StringBuilder(Revision.PROP_RESOURCE).append('.').
185                append(Resource.PROP_RESOURCEID).toString(),
186                resourceIds.toArray(new String[0])),
187                QueryParameterFactory.<Boolean>getEqualPropertyParam(
188                Revision.PROP_HEADREVISION, true)));
189            commitWriter.save(commitDomain);
190            if (currentHeads != null && !currentHeads.isEmpty()) {
191                for (Revision revision : currentHeads) {
192                    revision.setHeadRevision(false);
193                }
194                revisionWriter.update(currentHeads.toArray(new Revision[0]));
195            }
196        }

198        public Set<com.smartitengineering.version.api.Commit> searchForCommits(
199            Collection<QueryParameter> parameters) {
200            Collection<QueryParameter> params = parameters == null ? Collections.<QueryParameter>emptyList()
201                : parameters;
202            List<Commit> commits = commitReader.getList(params.toArray(
203                new QueryParameter[0]));
204            LinkedHashSet<com.smartitengineering.version.api.Commit> result =
205                new LinkedHashSet<com.smartitengineering.version.api.Commit>();
206            if (commits != null) {
207                for (Commit commit : commits) {
208                    result.add(MetaFactory.transformMetaCommit(commit));
209                }
210            }
211            return result;
212        }
