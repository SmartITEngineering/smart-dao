019    package com.smartitengineering.version.impl.jgit.domain;
020    
021    import com.smartitengineering.domain.AbstractPersistentDTO;
022    import com.smartitengineering.domain.PersistentDTO;
023    import org.apache.commons.lang.StringUtils;
024    
025    /**
026     *
027     * @author imyousuf
028     */
029    public class Revision
030        extends AbstractPersistentDTO<Revision>
031        implements PersistentDTO<Revision> {
032    
033        public static final String PROP_RESOURCE = "resource";
034        public static final String PROP_REVISIONID = "revisionId";
035        public static final String PROP_HEADREVISION = "headRevision";
036        public static final String PROP_COMMIT = "commit";
037        protected Resource resource;
038        protected String revisionId;
039        protected boolean headRevision;
040        protected Commit commit;
041    
042        /**
043         * Get the value of commit
044         *
045         * @return the value of commit
046         */
047        public Commit getCommit() {
048            return commit;
049        }
050    
051        /**
052         * Set the value of commit
053         *
054         * @param commit new value of commit
055         */
056        public void setCommit(Commit commit) {
057            this.commit = commit;
058        }
059    
060        /**
061         * Get the value of headRevision
062         *
063         * @return the value of headRevision
064         */
065        public boolean isHeadRevision() {
066            return headRevision;
067        }
068    
069        /**
070         * Set the value of headRevision
071         *
072         * @param headRevision new value of headRevision
073         */
074        public void setHeadRevision(boolean headRevision) {
075            this.headRevision = headRevision;
076        }
077    
078        /**
079         * Get the value of revisionId
080         *
081         * @return the value of revisionId
082         */
083        public String getRevisionId() {
084            return revisionId;
085        }
086    
087        /**
088         * Set the value of revisionId
089         *
090         * @param revisionId new value of revisionId
091         */
092        public void setRevisionId(String revisionId) {
093            this.revisionId = revisionId;
094        }
095    
096        /**
097         * Get the value of resource
098         *
099         * @return the value of resource
100         */
101        public Resource getResource() {
102            return resource;
103        }
104    
105        /**
106         * Set the value of resource
107         *
108         * @param resource new value of resource
109         */
110        public void setResource(Resource resource) {
111            this.resource = resource;
112        }
113    
114        public boolean isValid() {
115            if (StringUtils.isBlank(revisionId) || resource == null || !resource.
116                isValid()) {
117                return false;
118            }
119            if(commit == null) {
120                return false;
121            }
122            return true;
123        }
124    }
