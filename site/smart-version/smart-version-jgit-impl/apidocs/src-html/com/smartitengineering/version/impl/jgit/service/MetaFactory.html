019    package com.smartitengineering.version.impl.jgit.service;
020    
021    import com.smartitengineering.util.bean.BeanFactoryRegistrar;
022    import com.smartitengineering.util.bean.annotations.Aggregator;
023    import com.smartitengineering.util.bean.annotations.InjectableField;
024    import com.smartitengineering.version.api.factory.VersionAPI;
025    import com.smartitengineering.version.impl.jgit.domain.Author;
026    import com.smartitengineering.version.impl.jgit.domain.Commit;
027    import com.smartitengineering.version.impl.jgit.domain.Resource;
028    import com.smartitengineering.version.impl.jgit.domain.Revision;
029    import java.util.LinkedHashSet;
030    
031    /**
032     *
033     * @author imyousuf
034     */
035    @Aggregator(contextName = "com.smartitnengineering.smart-dao.smart-version-jgit")
036    public class MetaFactory {
037    
038        private static MetaFactory factory;
039        @InjectableField(beanName = "metaRCSConfig")
040        private RCSConfig config;
041        @InjectableField
042        private MetaRCSService metaRCSService;
043    
044        public RCSConfig getConfig() {
045            if (config == null) {
046                config = new RCSConfig();
047                config.setAllowNoChangeCommit(false);
048                config.setConcurrentWriteOperations(1);
049            }
050            return config;
051        }
052    
053        public MetaRCSService getMetaRCSService() {
054            return metaRCSService;
055        }
056    
057        private MetaFactory() {
058            BeanFactoryRegistrar.aggregate(this);
059        }
060    
061        public static final MetaFactory getInstance() {
062            if (factory == null) {
063                factory = new MetaFactory();
064            }
065            return factory;
066        }
067    
068        public static Commit transformAPICommit(
069            final com.smartitengineering.version.api.Commit apiCommit) {
070            Commit commit = new Commit();
071            commit.setCommitId(apiCommit.getCommitId());
072            commit.setCommitMessage(apiCommit.getCommitMessage());
073            Author committer = new Author();
074            committer.setName(apiCommit.getAuthor().getName());
075            committer.setEmail(apiCommit.getAuthor().getEmail());
076            commit.setCommitter(committer);
077            commit.setDateTime(apiCommit.getCommitTime());
078            commit.setParentCommitId(apiCommit.getParentCommitId());
079            for (com.smartitengineering.version.api.Revision apiRevision : apiCommit.
080                getRevisions()) {
081                commit.getRevisions().add(transformAPIRevision(commit, apiRevision));
082            }
083            return commit;
084        }
085    
086        public static Revision transformAPIRevision(
087            final Commit commit,
088            final com.smartitengineering.version.api.Revision apiRevision) {
089            Revision revision = new Revision();
090            revision.setCommit(commit);
091            Resource resource = new Resource();
092            resource.setResourceId(apiRevision.getResource().getId());
093            resource.setDeleted(apiRevision.getResource().isDeleted());
094            resource.setMimeType(apiRevision.getResource().getMimeType());
095            revision.setResource(resource);
096            revision.setRevisionId(apiRevision.getRevisionId());
097            return revision;
098        }
099    
100        public static com.smartitengineering.version.api.Commit transformMetaCommit(
101            Commit commit) {
102            com.smartitengineering.version.api.Author apiAuthor = VersionAPI.
103                createAuthor(commit.getCommitter().getName(), commit.getCommitter().
104                getEmail());
105            LinkedHashSet<com.smartitengineering.version.api.Revision> apiRevisions =
106                new LinkedHashSet<com.smartitengineering.version.api.Revision>(commit.getRevisions().
107                size());
108            for (Revision revision : commit.getRevisions()) {
109                apiRevisions.add(transformMetaRevision(revision));
110            }
111            return VersionAPI.createCommit(apiRevisions, commit.getCommitId(),
112                commit.getParentCommitId(), commit.getCommitMessage(), apiAuthor,
113                commit.getDateTime());
114        }
115    
116        public static com.smartitengineering.version.api.Revision transformMetaRevision(
117            Revision revision) {
118            final com.smartitengineering.version.api.Resource apiResource;
119            if (VersionAPI.getInstance().getVersionControlReadDao() == null) {
120                apiResource = VersionAPI.createResource(revision.getResource().
121                    getResourceId(), "", revision.getResource().isDeleted(),
122                    revision.getResource().getMimeType());
123            }
124            else {
125                final com.smartitengineering.version.api.Resource apiResourceFromApi =
126                    VersionAPI.getInstance().getVersionControlReadDao().
127                    getResourceByRevision(
128                    revision.getRevisionId(), revision.getResource().getResourceId());
129                apiResource = VersionAPI.createResource(revision.getResource().
130                    getResourceId(), apiResourceFromApi.getContent(),
131                    revision.getResource().isDeleted(), revision.getResource().
132                    getMimeType());
133            }
134            return VersionAPI.createRevision(apiResource, revision.getRevisionId());
135        }
136    }
