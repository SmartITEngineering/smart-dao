085        public void init()
086            throws IOException {
087            if (initialized) {
088                throw new IllegalStateException("Impl already initialized");
089            }
090            if (StringUtils.isBlank(getReadRepositoryLocation()) || StringUtils.
091                isBlank(getWriteRepositoryLocation())) {
092                throw new IllegalStateException("Repository location not set!");
093            }
094            File writeRepoDir = new File(getWriteRepositoryLocation());
095            writeRepository = new Repository(writeRepoDir);
096            if (!writeRepoDir.exists()) {
097                writeRepository.create();
098            }
099            writeRepository.close();
100            File readRepoDir = new File(getReadRepositoryLocation());
101            readRepository = new Repository(readRepoDir);
102            if (!readRepoDir.exists()) {
103                readRepository.create();
104            }
105            readRepository.close();
106            executorService = Executors.newFixedThreadPool(getConfig().
107                getConcurrentWriteOperations());
108            initialized = true;
109        }
110    
111        public void finish() {
112            if (writeRepository != null) {
113                writeRepository.close();
114            }
115            if (readRepository != null) {
116                readRepository.close();
117            }
118        }

191        public void store(final Commit commit,
192                          final WriterCallback callback) {
193            executorService.submit(new Runnable() {
194    
195                public void run() {
196                    WriteStatus status = null;
197                    String comment = null;
198                    Throwable error = null;
199                    try {
200                        Tree head = getHeadTree(writeRepository);
201                        addOrUpdateToHead(commit, head);
202                        prepareCommit(head, commit);
203                        status = WriteStatus.STORE_PASS;
204                        comment = "OK";
205                        error = null;
206                    }
207                    catch (Throwable ex) {
208                        status = WriteStatus.STORE_FAIL;
209                        comment = ex.getMessage();
210                        error = ex;
211                        throw new RuntimeException(ex);
212                    }
213                    finally {
214                        if (callback != null) {
215                            callback.handle(commit, status, comment, error);
216                        }
217                        try {
218                            reInitReadRepository();
219                        }
220                        catch (IOException ex) {
221                            ex.printStackTrace();
222                        }
223                    }
224                }
225            });
226        }

228        public VersionedResource getVersionedResource(final String resourceId) {
229            try {
230                String trimmedResourceId = VersionAPI.trimToPropertResourceId(
231                    resourceId);
232                if (StringUtils.isBlank(trimmedResourceId)) {
233                    throw new IllegalArgumentException("Invalid resource id!");
234                }
235                Set<ObjectId> revisionIds = getGraphForResourceId(trimmedResourceId);
236                if (revisionIds == null || revisionIds.isEmpty()) {
237                    throw new IllegalArgumentException("Resource id doesn't exist!");
238                }
239                Revision[] revisions = new Revision[revisionIds.size()];
240                int i = 0;
241                for (ObjectId revisionId : revisionIds) {
242                    String revisionIdStr = ObjectId.toString(revisionId);
243                    String content = new String(readObject(revisionIdStr));
244                    revisions[i++] = VersionAPI.createRevision(VersionAPI.
245                        createResource(trimmedResourceId, content), revisionIdStr);
246                }
247                return VersionAPI.createVersionedResource(Arrays.asList(revisions));
248            }
249            catch (Throwable ex) {
250                throw new RuntimeException(ex);
251            }
252        }

254        public Resource getResource(final String resourceId) {
255            try {
256                String trimmedResourceId = VersionAPI.trimToPropertResourceId(
257                    resourceId);
258                if (StringUtils.isBlank(trimmedResourceId)) {
259                    throw new IllegalArgumentException("Invalid resource id!");
260                }
261                ObjectId resourceObjectId;
262                Tree head = getHeadTree(getReadRepository());
263                if (!head.existsBlob(trimmedResourceId)) {
264                    throw new IllegalArgumentException("Resource id doesn't exist!");
265                }
266                TreeEntry treeEntry = head.findBlobMember(trimmedResourceId);
267                resourceObjectId = treeEntry.getId();
268                return VersionAPI.createResource(trimmedResourceId, new String(
269                    readObject(ObjectId.toString(resourceObjectId))));
270            }
271            catch (Throwable ex) {
272                throw new RuntimeException(ex);
273            }
274        }

294        public byte[] readObject(final String objectIdStr)
295            throws IOException,
296                   IllegalArgumentException {
297            if (StringUtils.isBlank(objectIdStr)) {
298                throw new IllegalArgumentException("Invalid Object id!");
299            }
300            ObjectId objectId = ObjectId.fromString(objectIdStr);
301            ObjectLoader objectLoader = getReadRepository().openObject(objectId);
302            if (objectLoader.getType() != Constants.OBJ_BLOB) {
303                throw new IllegalArgumentException("Not a blob: " + objectIdStr);
304            }
305            return objectLoader.getBytes();
306        }

318        public Map<String, byte[]> readBlobObjects(final String... objectIds)
319            throws IOException,
320                   IllegalArgumentException {
321            checkInitialized();
322            if (objectIds == null || objectIds.length <= 0) {
323                throw new IllegalArgumentException("Empty Object IDs!");
324            }
325            Map<String, byte[]> blobs =
326                new HashMap<String, byte[]>(objectIds.length);
327            for (String objectIdStr : objectIds) {
328                byte[] bytes = readObject(objectIdStr);
329                blobs.put(objectIdStr, bytes);
330            }
331            return blobs;
332        }

334        protected Tree getHeadTree(Repository repository)
335            throws IOException {
336            Tree head;
337            org.spearce.jgit.lib.Commit headCommit = repository.mapCommit(
338                Constants.HEAD);
339            if (headCommit == null) {
340                head = new Tree(repository);
341            }
342            else {
343                head = headCommit.getTree();
344            }
345            return head;
346        }

348        protected ObjectId addOrUpdateToHead(final Commit commit,
349                                             final Tree head)
350            throws IOException {
351            for (Revision revision : commit.getRevisions()) {
352                String objectPath = revision.getResource().getId();
353                FileTreeEntry treeEntry;
354                boolean newEntry = false;
355                if (head.existsBlob(objectPath)) {
356                    treeEntry = (FileTreeEntry) head.findBlobMember(objectPath);
357                }
358                else {
359                    treeEntry = head.addFile(objectPath);
360                    newEntry = true;
361                }
362                treeEntry.setExecutable(false);
363                if (!revision.getResource().isDeleted()) {
364                    if (revision instanceof MutableRevision) {
365                        ObjectId revisionId = getObjectWriter().writeBlob(
366                            revision.getResource().getContentSize(),
367                            revision.getResource().getContentAsStream());
368                        MutableRevision mutableRevision = (MutableRevision) revision;
369                        mutableRevision.setRevisionId(ObjectId.toString(revisionId));
370                        treeEntry.setId(revisionId);
371                    }
372                    else {
373                        throw new IllegalArgumentException(
374                            "SPI not implemented by API!");
375                    }
376                }
377                else if (!newEntry) {
378                    if (revision instanceof MutableRevision) {
379                        ObjectId revisionId = treeEntry.getId();
380                        MutableRevision mutableRevision = (MutableRevision) revision;
381                        mutableRevision.setRevisionId(ObjectId.toString(revisionId));
382                        treeEntry.delete();
383                    }
384                    else {
385                        throw new IllegalArgumentException(
386                            "SPI not implemented by API!");
387                    }
388                }
389            }
390            GitIndex index = getWriteRepository().getIndex();
391            index.readTree(head);
392            index.write();
393            ObjectId newHeadId = index.writeTree();
394            head.setId(newHeadId);
395            return newHeadId;
396        }

405        protected void performCommit(final Commit newCommit,
406                                     final Tree head)
407            throws IOException {
408            ObjectId[] parentIds;
409            ObjectId currentHeadId = getWriteRepository().resolve(Constants.HEAD);
410            if (currentHeadId != null) {
411                parentIds = new ObjectId[]{currentHeadId};
412            }
413            else {
414                parentIds = new ObjectId[0];
415            }
416            org.spearce.jgit.lib.Commit commit = new org.spearce.jgit.lib.Commit(
417                getWriteRepository(), parentIds);
418            commit.setTree(head);
419            commit.setTreeId(head.getId());
420            PersonIdent person = new PersonIdent(newCommit.getAuthor().getName(),
421                newCommit.getAuthor().getEmail());
422            commit.setAuthor(person);
423            commit.setCommitter(person);
424            commit.setMessage(newCommit.getCommitMessage());
425            ObjectId newCommitId = getObjectWriter().writeCommit(commit);
426            if (newCommit instanceof MutableCommit) {
427                MutableCommit mutableCommit = (MutableCommit) newCommit;
428                mutableCommit.setCommitId(ObjectId.toString(newCommitId));
429                mutableCommit.setCommitTime(commit.getCommitter().getWhen());
430                commit.setCommitId(newCommitId);
431                if (commit.getParentIds().length > 0) {
432                    mutableCommit.setParentCommitId(ObjectId.toString(commit.
433                        getParentIds()[0]));
434                }
435                else {
436                    mutableCommit.setParentCommitId(ObjectId.toString(ObjectId.
437                        zeroId()));
438                }
439            }
440            else {
441                throw new IllegalArgumentException("SPI not implemented by API!");
442            }
443            RefUpdate refUpdate =
444                getWriteRepository().updateRef(Constants.HEAD);
445            refUpdate.setNewObjectId(commit.getCommitId());
446            refUpdate.setRefLogMessage(commit.getMessage(), false);
447            refUpdate.forceUpdate();
448        }
449    
450        protected void prepareCommit(final Tree head,
451                                     final Commit commit)
452            throws IOException,
453                   IOException {
454            ObjectId currentHeadId =
455                getWriteRepository().resolve(Constants.HEAD);
456            boolean commitAvailable = true;
457            if (currentHeadId != null) {
458                org.spearce.jgit.lib.Commit headCommit =
459                    writeRepository.mapCommit(currentHeadId);
460                if (headCommit != null) {
461                    Tree headTree = headCommit.getTree();
462                    if (headTree != null && head.getId().equals(headTree.getId())) {
463                        commitAvailable = false;
464                    }
465                }
466            }
467            if (commitAvailable || getConfig().isAllowNoChangeCommit()) {
468                performCommit(commit, head);
469            }
470        }

472        protected Set<ObjectId> getGraphForResourceId(String resourceId)
473            throws MissingObjectException,
474                   IncorrectObjectTypeException,
475                   IOException {
476            final Set<ObjectId> versions = new LinkedHashSet<ObjectId>();
477            final RevWalk rw = new RevWalk(getReadRepository());
478            final TreeWalk tw = new TreeWalk(getReadRepository());
479            rw.markStart(rw.parseCommit(getReadRepository().resolve(Constants.HEAD)));
480            tw.setFilter(TreeFilter.ANY_DIFF);
481            RevCommit c;
482            while ((c = rw.next()) != null) {
483                final ObjectId[] p = new ObjectId[c.getParentCount() + 1];
484                for (int i = 0; i < c.getParentCount(); i++) {
485                    rw.parse(c.getParent(i));
486                    p[i] = c.getParent(i).getTree();
487                }
488                final int me = p.length - 1;
489                p[me] = c.getTree();
490                tw.reset(p);
491                while (tw.next()) {
492                    if (tw.getFileMode(me).getObjectType() == Constants.OBJ_BLOB) {
493                        // This path was modified relative to the ancestor(s)
494                        String s = tw.getPathString();
495                        if (s != null && s.equals(resourceId)) {
496                            versions.add(tw.getObjectId(me));
497                        }
498                    }
499                    if (tw.isSubtree()) {
500                        // make sure we recurse into modified directories
501                        tw.enterSubtree();
502                    }
503                }
504            }
505            return versions;
506        }
