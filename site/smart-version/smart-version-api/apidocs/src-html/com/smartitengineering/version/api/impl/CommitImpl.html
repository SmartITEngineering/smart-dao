019    package com.smartitengineering.version.api.impl;
020    
021    import com.smartitengineering.version.api.Author;
022    import com.smartitengineering.version.api.Commit;
023    import com.smartitengineering.version.api.Revision;
024    import com.smartitengineering.version.api.spi.MutableCommit;
025    import java.util.Collection;
026    import java.util.Collections;
027    import java.util.Date;
028    import org.apache.commons.lang.StringUtils;
029    
030    /**
031     * Default implementation of commit
032     * @author imyousuf
033     */
034    public class CommitImpl
035        implements Commit,
036                   MutableCommit {
037    
038        private Collection<Revision> revisions;
039        private String commitId;
040        private String commitMessage;
041        private Author author;
042        private String parentCommitId;
043        protected Date commitTime;
044    
045        /**
046         * Get the value of commitTime
047         *
048         * @return the value of commitTime
049         */
050        public Date getCommitTime() {
051            return commitTime;
052        }
053    
054        /**
055         * Set the value of commitTime
056         *
057         * @param commitTime new value of commitTime
058         */
059        public void setCommitTime(Date commitTime) {
060            this.commitTime = commitTime;
061        }
062    
063        /**
064         * Get the value of commitParentId
065         *
066         * @return the value of commitParentId
067         */
068        public String getParentCommitId() {
069            return parentCommitId;
070        }
071    
072        /**
073         * Set the value of commitParentId
074         *
075         * @param commitParentId new value of commitParentId
076         */
077        public void setParentCommitId(String parentCommitId) {
078            this.parentCommitId = parentCommitId;
079        }
080    
081        /**
082         * Get the value of author
083         *
084         * @return the value of author
085         */
086        public Author getAuthor() {
087            if (author == null) {
088                throw new IllegalStateException(
089                    "Author must be set before attempting to retrieve it!");
090            }
091            return author;
092        }
093    
094        /**
095         * Set the value of author
096         *
097         * @param author new value of author
098         */
099        public void setAuthor(Author author) {
100            if (author == null) {
101                return;
102            }
103            this.author = author;
104        }
105    
106        /**
107         * Get the value of commitMessage
108         *
109         * @return the value of commitMessage
110         */
111        public String getCommitMessage() {
112            if (StringUtils.isBlank(commitMessage)) {
113                throw new IllegalStateException(
114                    "Commit message (unblank) must be set before attempting to retrieve it!");
115            }
116            return commitMessage;
117        }
118    
119        /**
120         * Set the value of commitMessage
121         *
122         * @param commitMessage new value of commitMessage
123         */
124        public void setCommitMessage(String commitMessage) {
125            if (StringUtils.isBlank(commitMessage)) {
126                return;
127            }
128            this.commitMessage = commitMessage;
129        }
130    
131        /**
132         * Get the value of commitId
133         *
134         * @return the value of commitId
135         */
136        public String getCommitId() {
137            return commitId;
138        }
139    
140        /**
141         * Set the value of commitId
142         *
143         * @param commitId new value of commitId
144         */
145        public void setCommitId(String commitId) {
146            this.commitId = commitId;
147        }
148    
149        /**
150         * Get the value of revisions
151         *
152         * @return the value of revisions
153         */
154        public Collection<Revision> getRevisions() {
155            if (revisions == null) {
156                return Collections.emptySet();
157            }
158            return revisions;
159        }
160    
161        /**
162         * Set the value of revisions
163         *
164         * @param revisions new value of revisions
165         */
166        public void setRevisions(Collection<Revision> revisions) {
167            this.revisions = revisions;
168        }
169    
170        @Override
171        public boolean equals(Object obj) {
172            if (obj instanceof Commit) {
173                Commit commitObj = (Commit) obj;
174                if (StringUtils.isNotBlank(getCommitId())) {
175                    return getCommitId().equals(commitObj.getCommitId());
176                }
177            }
178            return super.equals(obj);
179        }
180    
181        @Override
182        public int hashCode() {
183            if (StringUtils.isNotBlank(getCommitId())) {
184                return getCommitId().hashCode();
185            }
186            return super.hashCode();
187        }
188    }
