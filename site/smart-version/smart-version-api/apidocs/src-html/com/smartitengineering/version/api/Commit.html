019    package com.smartitengineering.version.api;
020    
021    import java.util.Collection;
022    import java.util.Date;
023    
024    /**
025     * It represents a change in the system.
026     * @author imyousuf
027     */
028    public interface Commit {
029    
030        /**
031         * Get the revisions that have changed in the commit.
032         * @return Changes of the commit
033         */
034        public Collection<Revision> getRevisions();
035    
036        /**
037         * Retrieves the unique identifier for this commit
038         * @return Unique commit identifier
039         */
040        public String getCommitId();
041    
042        /**
043         * Retrieve the commit log message for the change.
044         * @return Commit log
045         */
046        public String getCommitMessage();
047    
048        /**
049         * Retrieves the Author of the commit. It could be useful to check who has
050         * performed different updates to a set of resources
051         * @return Committer
052         */
053        public Author getAuthor();
054    
055        /**
056         * Retrieves the date-time stamp of when the commit was performed.
057         * @return Date-time of the commit
058         */
059        public Date getCommitTime();
060    
061        /**
062         * Get the parent commit id of the current commit.
063         * @return Parent commit's id or blank string if root commit
064         */
065        public String getParentCommitId();
066    }
