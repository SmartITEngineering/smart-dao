019    package com.smartitengineering.version.api.dao;
020    
021    import com.smartitengineering.dao.common.queryparam.QueryParameter;
022    import com.smartitengineering.version.api.Commit;
023    import com.smartitengineering.version.api.Resource;
024    import com.smartitengineering.version.api.Revision;
025    import com.smartitengineering.version.api.VersionedResource;
026    import java.util.Collection;
027    
028    /**
029     * Its the main interface through which all CRUD opertation to the RCS will be
030     * made. Service providers will mainly implement this and provide access to
031     * users.
032     * @author imyousuf
033     */
034    public interface VersionControlReadDao {
035    
036        /**
037         * Get a resource with its versions specified by the resource id.
038         * @param resourceId Unique identifier of the resource
039         * @return The versioned representation of the resource.
040         */
041        public VersionedResource getVersionedResource(final String resourceId);
042    
043        /**
044         * Retrieve the HEAD version of the specified resource id.
045         * @param resourceId The resource's id to fetch
046         * @return Fetched resource for id
047         */
048        public Resource getResource(final String resourceId);
049    
050        /**
051         * Retrieve the specified revision of the specified resource
052         * @param revisionId The revision to get
053         * @param resourceId The resource to get
054         * @return The resource with the specified resource id and revision id
055         */
056        public Resource getResourceByRevision(final String revisionId,
057                                              final String resourceId);
058    
059        /**
060         * Search for commits. Please implementation documentation to check whether
061         * this operation is supported or not and how to use it.
062         * @param parameters Parameters to be used for search
063         * @return Search result or empty collection if nothing could be matched
064         */
065        public Collection<Commit> searchForCommits(
066            final Collection<QueryParameter> parameters);
067    
068        /**
069         * Search for VersionedObjects. Please implementation documentation to check
070         * whether this operation is supported or not and how to use it.
071         * @param parameters Parameters to be used fo search
072         * @return Search result or empty collection if nothing could be matched
073         */
074        public Collection<Revision> searchForRevisions(
075            final Collection<QueryParameter> parameters);
076    }
