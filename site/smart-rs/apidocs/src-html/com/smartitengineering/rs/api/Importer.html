019    package com.smartitengineering.rs.api;
020    
021    import java.io.IOException;
022    import java.io.InputStream;
023    import java.lang.annotation.Annotation;
024    import java.lang.reflect.Type;
025    import java.util.List;
026    import java.util.Map;
027    import javax.ws.rs.core.MediaType;
028    
029    /**
030     * This API provides SPIs to use it to import objects from stream of media type
031     * it specifies. It is the responsibility of the API client to invoke it for
032     * appropriate {@link MediaType}. API implementors should explicitly mention the
033     * media type it supports. API implementors should also ensure thread safety.
034     * @author imyousuf
035     * @since 0.4
036     */
037    public interface Importer {
038    
039        /**
040         * An operation that checks whether the the object type qualified by its
041         * class is supported by this importer. This operation should be consulted
042         * before invoking/using importObject.
043         * @param clazz The class quantifying the object type
044         * @param genericType The generic applied to the object, if any
045         * @param annotations Annotations used on the class
046         * @return True if and only if import is supported else false.
047         */
048        boolean isImportSupported(Class clazz,
049                                  Type genericType,
050                                  Annotation[] annotations);
051    
052        /**
053         * Import the object quantified by clazz and genericType and qualified by
054         * the annotaitons from the inpuStream and use the headers to retrieve meta
055         * informations regarding the object.
056         * @param clazz The class of the object
057         * @param genericType The generic used with the object.
058         * @param annotations Annotations qualifying the object's class.
059         * @param inputStream The input stream to read and parse the object from.
060         * @param headers The headers that holds the meta information
061         * @return The object formed by from the input stream. Should never be NULL.
062         * @throws java.io.IOException If any error in reading from the stream, i.e.
063         *                             either format error or any other error.
064         */
065        Object importObject(Class clazz,
066                            Type genericType,
067                            Annotation[] annotations,
068                            InputStream inputStream,
069                            Map<Object, List<Object>> headers)
070            throws IOException;
071    
072        /**
073         * Provides the {@link MediaType} that is supported by this importer.
074         * @return The supported media type
075         */
076        MediaType getMediaType();
077    }
