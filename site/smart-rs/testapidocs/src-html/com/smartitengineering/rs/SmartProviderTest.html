074        public void testSetters() {
075            assertNotNull(smartProvider.getExporters());
076            assertNotNull(smartProvider.getImporters());
077            context.checking(new Expectations() {
078    
079                {
080                    exactly(1).of(mockExporter).getMediaType();
081                    will(returnValue(null));
082                    exactly(2).of(mockExporter).getMediaType();
083                    will(returnValue(MediaType.APPLICATION_XML_TYPE.toString()));
084                }
085            });
086            smartProvider.setExporter(mockExporter);
087            assertEquals(smartProvider.getExporters().size(), 0);
088            smartProvider.setExporter(mockExporter);
089            assertEquals(smartProvider.getExporters().size(), 1);
090            smartProvider.setExporter(null);
091            assertEquals(smartProvider.getExporters().size(), 1);
092            context.checking(new Expectations() {
093    
094                {
095                    exactly(1).of(mockImporter).getMediaType();
096                    will(returnValue(null));
097                    exactly(2).of(mockImporter).getMediaType();
098                    will(returnValue(MediaType.APPLICATION_XML_TYPE.toString()));
099                }
100            });
101            smartProvider.setImporter(mockImporter);
102            assertEquals(smartProvider.getImporters().size(), 0);
103            smartProvider.setImporter(mockImporter);
104            assertEquals(smartProvider.getImporters().size(), 1);
105            smartProvider.setImporter(null);
106            assertEquals(smartProvider.getImporters().size(), 1);
107        }

109        public void testIsWriteable() {
110            initializeProvider();
111            context.checking(new Expectations() {
112    
113                {
114                    exactly(2).of(mockExporter).getMediaType();
115                    will(returnValue(MediaType.APPLICATION_JSON_TYPE.toString()));
116                }
117            });
118            smartProvider.setExporter(mockExporter);
119            assertEquals(2, smartProvider.getExporters().size());
120            context.checking(new Expectations() {
121    
122                {
123                    exactly(1).of(mockExporter).isExportSupported(with(any(
124                        Class.class)), with(aNull(Type.class)), (Annotation[]) with(
125                        anything()));
126                    will(returnValue(true));
127                    exactly(1).of(mockExporter).isExportSupported(with(any(
128                        Class.class)), with(aNull(Type.class)), (Annotation[]) with(
129                        anything()));
130                    will(returnValue(false));
131                }
132            });
133            assertTrue(smartProvider.isWriteable(SmartProvider.class, null,
134                new Annotation[]{}, MediaType.APPLICATION_XML_TYPE));
135            assertFalse(smartProvider.isWriteable(SmartProvider.class, null,
136                new Annotation[]{}, MediaType.APPLICATION_JSON_TYPE));
137            assertFalse(smartProvider.isWriteable(SmartProvider.class, null,
138                new Annotation[]{}, MediaType.APPLICATION_ATOM_XML_TYPE));
139        }

164        public void testWriteTo() {
165            initializeProvider();
166            context.checking(new Expectations() {
167    
168                {
169                    try {
170                        exactly(1).of(mockExporter).
171                            isExportSupported(with(any(Class.class)), with(aNull(
172                            Type.class)), (Annotation[]) with(anything()));
173                        will(returnValue(true));
174                        exactly(1).of(mockExporter).exportObject(with(any(
175                            Importer.class)), with(any(Class.class)), with(aNull(
176                            Type.class)), (Annotation[]) with(anything()), with(any(
177                            OutputStream.class)), with(any(Map.class)));
178                    }
179                    catch (IOException ex) {
180                        ex.printStackTrace();
181                    }
182                }
183            });
184            MultivaluedMap map = context.mock(MultivaluedMap.class);
185            try {
186                smartProvider.writeTo(mockImporter,
187                    SmartProvider.class, null, new Annotation[0],
188                    MediaType.APPLICATION_XML_TYPE, map, new ByteArrayOutputStream());
189            }
190            catch (IOException ex) {
191                fail(ex.getMessage());
192            }
193            catch (WebApplicationException ex) {
194                fail(ex.getMessage());
195            }
196            try {
197                smartProvider.writeTo(mockImporter,
198                    SmartProvider.class, null, new Annotation[0],
199                    MediaType.APPLICATION_JSON_TYPE, map,
200                    new ByteArrayOutputStream());
201                fail("Should have failed!");
202            }
203            catch (IOException ex) {
204                assertEquals("Export is supported for this object!", ex.getMessage());
205            }
206            catch (WebApplicationException ex) {
207                fail(ex.getMessage());
208            }
209        }

244        public void testReadFrom() {
245            initializeProvider();
246            context.checking(new Expectations() {
247    
248                {
249                    try {
250                        exactly(1).of(mockImporter).isImportSupported(with(any(
251                            Class.class)), with(aNull(Type.class)),
252                            (Annotation[]) with(anything()));
253                        will(returnValue(true));
254                        exactly(1).of(mockImporter).importObject(with(any(
255                            Class.class)), with(aNull(Type.class)),
256                            (Annotation[]) with(anything()), with(any(
257                            InputStream.class)), with(any(Map.class)));
258                        will(returnValue("Object"));
259                    }
260                    catch (IOException ex) {
261                        ex.printStackTrace();
262                    }
263                }
264            });
265            MultivaluedMap map = context.mock(MultivaluedMap.class);
266            try {
267                Object object = smartProvider.readFrom(SmartProvider.class, null,
268                    new Annotation[0], MediaType.APPLICATION_XML_TYPE, map,
269                    new ByteArrayInputStream(new byte[10]));
270                assertEquals("Object", object);
271            }
272            catch (IOException ex) {
273                fail(ex.getMessage());
274            }
275            catch (WebApplicationException ex) {
276                fail(ex.getMessage());
277            }
278            try {
279                smartProvider.readFrom(SmartProvider.class, null, new Annotation[0],
280                    MediaType.APPLICATION_JSON_TYPE, map,
281                    new ByteArrayInputStream(new byte[10]));
282                fail("Should have failed!");
283            }
284            catch (IOException ex) {
285                assertEquals("Import is not supported for this object!", ex.
286                    getMessage());
287            }
288            catch (WebApplicationException ex) {
289                fail(ex.getMessage());
290            }
291        }

305            smartProvider.setImporter(mockImporter);
