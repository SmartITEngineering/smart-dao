025    /**
026     * This interface will basically help you undertand the top-level type of the
027     * query parameter and cast it. Its intended so that end user can easily
028     * determine how to use them.
029     * @author imyousuf
030     */
031    public interface QueryParameterCastHelper<T extends QueryParameter> {
032    
033        /**
034         * Returns the parameter types of the helper
035         * @return Type of the parameters it supports
036         */
037        public Collection<ParameterType> getParameterTypes();
038    
039        /**
040         * Checks whether the parameter is a param with operator or not
041         * @param parameter The parameter to check
042         * @return True if the parameter has operators
043         */
044        public boolean isWithOperator(QueryParameter parameter);
045    
046        /**
047         * Casts the parameter to parameter with operator.
048         * @param parameter The parameter to cast
049         * @return Casted version of the parameter
050         * @throws IllegalArgumentException If isWithOperator(parameter) return false
051         */
052        public QueryParameterWithOperator castToOperatorParam(
053            QueryParameter parameter)
054            throws IllegalArgumentException;
055    
056        /**
057         * Returns whether the cast helper supports any particular operator(s)
058         * @return true if the helper needs param to have certain operator(s)
059         */
060        public boolean hasOperator();
061    
062        /**
063         * Returns the supported operator types
064         * @return Supported operator types
065         * @throws UnsupportedOperationException If hasOperator returns false
066         */
067        public Collection<OperatorType> getOperators()
068            throws UnsupportedOperationException;
069    
070        /**
071         * Cast and return the parameter to the top level QueryParameter descendent
072         * if and only if the helper supports the parameter and operator
073         * @param parameter QueryParameter to be casted
074         * @return top level cast of parameter
075         * @throws IllegalArgumentException If parameter and operator of
076         *                                  QueryParameter is not supported by this
077         *                                  helper
078         */
079        public T cast(final QueryParameter parameter)
080            throws IllegalArgumentException;
