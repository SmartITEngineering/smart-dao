019    package com.smartitengineering.dao.common.queryparam;
020    
021    import java.util.Collections;
022    
023    /**
024     *
025     * @author imyousuf
026     */
027    public abstract class AbstractQueryParameterCastHelper<P extends QueryParameter>
028        implements QueryParameterCastHelper<P> {
029        
030        private boolean hasOperator;
031        
032        protected AbstractQueryParameterCastHelper(boolean hasOperator) {
033            this.hasOperator = hasOperator;
034        }
035    
036        protected AbstractQueryParameterCastHelper() {
037            this(false);
038        }
039    
040        public P cast(QueryParameter parameter)
041            throws IllegalArgumentException {
042            return (P) QueryParameterFactory.getCastedQueryParameter(
043                getEndTypeClass(), getParameterTypes(), hasOperator() ? getOperators()
044                : Collections.EMPTY_SET, parameter);
045        }
046    
047        public QueryParameterWithOperator castToOperatorParam(
048            QueryParameter parameter) {
049            if(!isWithOperator(parameter)) {
050                throw new IllegalArgumentException("Parameter does not support operators!");
051            }
052            return QueryParameterWithOperator.class.cast(parameter);
053        }
054    
055        public boolean isWithOperator(QueryParameter parameter) {
056            return parameter != null &&
057                parameter instanceof QueryParameterWithOperator;
058        }
059    
060        public boolean hasOperator() {
061            return hasOperator;
062        }
063    
064        public abstract Class<P> getEndTypeClass();
065    }
