262        public static List<QueryParameter<Void>> getMultiPropProjectionParam(
263            final String... propertyNames) {
264            List<QueryParameter<Void>> params = new ArrayList<QueryParameter<Void>>(
265                propertyNames.length);
266            for (String propertyName : propertyNames) {
267                params.add(getPropProjectionParam(propertyName));
268            }
269            return params;
270        }

272        public static <T> T getCastedQueryParameter(final Class<T> castClass,
273                                                    final Collection<ParameterType> parameterTypes,
274                                                    final Collection<OperatorType> operators,
275                                                    final QueryParameter parameter) {
276            if (parameterTypes.contains(parameter.getParameterType())) {
277                if (operators.isEmpty() ||
278                    (QueryParameterCastHelper.BI_OPERAND_PARAM_HELPER.isWithOperator(
279                    parameter) && operators.contains(
280                    QueryParameterCastHelper.BI_OPERAND_PARAM_HELPER.
281                    castToOperatorParam(parameter).getOperatorType()))) {
282                    return castClass.cast(parameter);
283                }
284            }
285            throw new IllegalArgumentException("Parameter type not supported");
286        }
