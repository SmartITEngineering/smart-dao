019    package com.smartitengineering.dao.impl.hibernate;
020    
021    import com.smartitengineering.dao.common.CommonDao;
022    import com.smartitengineering.dao.common.queryparam.QueryParameter;
023    import com.smartitengineering.dao.common.queryparam.QueryParameterFactory;
024    import com.smartitengineering.domain.PersistentDTO;
025    import java.util.Collection;
026    import java.util.Collections;
027    import java.util.HashSet;
028    import java.util.List;
029    import java.util.Set;
030    
031    /**
032     *
033     * @author Imran M Yousuf
034     */
035    public abstract class AbstractCommonDaoImpl<Template extends PersistentDTO>
036        extends AbstractDAO<Template>
037        implements CommonDao<Template> {
038    
039        private Class<? extends Template> entityClass;
040    
041        public void save(Template... states) {
042            if (states != null && states.length <= 0) {
043                return;
044            }
045            if (entityClass == null) {
046                entityClass = (Class<Template>) states[0].getClass();
047            }
048            createEntity(states);
049        }
050    
051        public void update(Template... states) {
052            if (states != null && states.length <= 0) {
053                return;
054            }
055            if (entityClass == null) {
056                entityClass = (Class<Template>) states[0].getClass();
057            }
058            updateEntity(states);
059        }
060    
061        public void delete(Template... states) {
062            if (states != null && states.length <= 0) {
063                return;
064            }
065            if (entityClass == null) {
066                entityClass = (Class<Template>) states[0].getClass();
067            }
068            deleteEntity(states);
069        }
070    
071        public Set<Template> getAll() {
072            return new HashSet<Template>(getList(Collections.<QueryParameter>
073                emptyList()));
074        }
075    
076        public Template getById(Integer id) {
077            QueryParameter<Integer> param = QueryParameterFactory.<Integer>
078                getEqualPropertyParam("id", id);
079            try {
080                return getSingle(param);
081            }
082            catch (Exception ex) {
083                ex.printStackTrace();
084                return null;
085            }
086        }
087    
088        public Set<Template> getByIds(List<Integer> ids) {
089            QueryParameter<Integer> param = QueryParameterFactory.<Integer>
090                getIsInPropertyParam("id", ids.toArray(new Integer[0]));
091            Collection<Template> result;
092            try {
093                result = getList(param);
094            }
095            catch (Exception ex) {
096                ex.printStackTrace();
097                result = Collections.<Template>emptyList();
098            }
099            return new HashSet<Template>(result);
100        }
101    
102        protected Class getEntityClass() {
103            return entityClass;
104        }
105    
106        protected void setEntityClass(Class<? extends Template> entityClass) {
107            this.entityClass = entityClass;
108        }
109    
110        public Template getSingle(List<QueryParameter> query) {
111            return readSingle(entityClass, query);
112        }
113    
114        public List<Template> getList(List<QueryParameter> query) {
115            return readList(entityClass, query);
116        }
117    
118        public <OtherTemplate extends Object> OtherTemplate getOther(
119            List<QueryParameter> query) {
120            return this.<OtherTemplate>readOther(entityClass, query);
121        }
122    
123        public Template getSingle(QueryParameter... query) {
124            return readSingle(entityClass, query);
125        }
126    
127        public List<Template> getList(QueryParameter... query) {
128            return readList(entityClass, query);
129        }
130    
131        public <OtherTemplate extends Object> OtherTemplate getOther(
132            QueryParameter... query) {
133            return this.<OtherTemplate>readOther(entityClass, query);
134        }
135    
136        public <OtherTemplate extends Object> List<OtherTemplate> getOtherList(
137            List<QueryParameter> query) {
138            return readOtherList(entityClass, query);
139        }
140    
141        public <OtherTemplate extends Object> List<OtherTemplate> getOtherList(
142            QueryParameter... query) {
143            return readOtherList(entityClass, query);
144        }
145    }
